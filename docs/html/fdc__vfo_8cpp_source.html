119double vfo_pid::calc(double pulse_pos) {
120
121 // Data pulse position adjustment == phase correction
122 double phase_error = m_cell_center - pulse_pos;
123 constexpr double phase_error_limitter = 0.25f;
124 phase_error = limit(phase_error, -m_cell_size_ref * phase_error_limitter,
125 m_cell_size_ref * phase_error_limitter);
126 double phase_err_diff = phase_error - m_prev_phase_error;
127 double phase_correction = (phase_error * 0.02f /*- phase_err_diff * 0.002f*/) * m_current_gain; // PD control (no I element)
128 constexpr double phase_correction_limitter = 0.125f;
129 phase_correction = limit(phase_correction, -m_cell_size_ref * phase_correction_limitter,
130 m_cell_size_ref * phase_correction_limitter);
131 pulse_pos += phase_correction;
132 m_prev_phase_error = phase_error;
133
134 // Cell size adjustment == frequency correction
135 double freq_error = m_cell_center - pulse_pos;
136 m_freq_bias += freq_error; // 'I' element
137 double new_cell_size = m_cell_size_ref - (m_freq_bias * 0.01f) * m_current_gain; // 'I' only control
138 set_cell_size(new_cell_size);
139
140 return pulse_pos;
141}
