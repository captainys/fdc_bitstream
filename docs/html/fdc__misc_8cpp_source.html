48std::vector<int> generate_interleaved_sector_list(size_t num_sector, int interleave = 1) {
49 std::vector<int> sector_list(num_sector, 0);
50 int pos = 0;
51 for (int i = 1; i <= num_sector; i++) {
52 while (sector_list[pos] != 0) {
53 if (++pos >= num_sector) pos = 0;
54 }
55 sector_list[pos] = i;
56 pos += interleave;
57 if (pos >= num_sector) pos = 0;
58 }
59 return sector_list;
60}

166std::vector<size_t> get_frequent_distribution(bit_array barray) {
167 barray.clear_wraparound_flag();
168 std::vector<size_t> freq_dist;
169 do {
170 size_t dist = barray.distance_to_next_pulse();
171 if(freq_dist.size() <= dist) {
172 freq_dist.resize(dist+1);
173 }
174 freq_dist[dist]++;
175 } while(!barray.is_wraparound());

179std::vector<size_t> find_peaks(const std::vector<size_t> &dist_freq) {
180 // LPF
181 std::vector<size_t> avg(dist_freq.size());
182 const int filt[5] { 1,2,3,2,1 };
183 const int flt_sum = 1+2+3+2+1;
184 for(int i = 0; i<dist_freq.size(); i++) {
185 size_t avg_tmp = 0;
186 for(int j=-2; j<=2; j++) {
187 size_t dt=0;
188 if (i+j < 0) {
189 dt = 0;
190 } else if(i+j >= dist_freq.size()) {
191 dt = 0;
192 } else {
193 dt = dist_freq[i+j];
194 }
195 avg_tmp += dt * filt[j+2];
196 }
197 avg[i] = avg_tmp / flt_sum;
198 }
199 //fdc_misc::display_histogram(avg); // for debug purpose
200
201 // detect peaks
202 std::vector<std::pair<size_t, size_t>> peaks;
203 bool flat_peak = false;
204 size_t lower_edge = 0;
205 for(size_t i=1; i<avg.size()-1; i++) {
206 if(avg[i-1] < avg[i] && avg[i+1] < avg[i]) { // a sharp peak
207 peaks.push_back(std::pair<size_t, size_t>(i, avg[i]));
208 flat_peak = false;
209 } else if(avg[i-1] < avg[i] && avg[i+1] == avg[i]) { // lower edge of a possible flat peak
210 flat_peak = true;
211 lower_edge = i;
212 } else if(avg[i-1] == avg[i] && avg[i+1] < avg[i] && flat_peak) { // upper edge of a flat peak
213 peaks.push_back(std::pair<size_t, size_t>((i+lower_edge)/2, avg[i]));
214 flat_peak = false;
215 } else if(avg[i-1] == avg[i] && avg[i+1] == avg[i] && flat_peak) { // top of flat peak
216 // nop
217 } else { // neither sharp peak, top of flat peak, nor edge of flat peak.
218 flat_peak = false;
219 }
220 }
221 if(flat_peak==true) {
222 size_t tmp = (avg.size()-1 + lower_edge) / 2;
223 peaks.push_back(std::pair<size_t, size_t>(tmp, avg[tmp]));
224 }
225
226 // sort peaks by value (frequency)
227 std::sort(peaks.begin(), peaks.end(), [](const std::pair<size_t, size_t> &left, const std::pair<size_t, size_t> &right) { return left.second > right.second; });
228
229 if (peaks.size()<3) {
230 for(size_t i=0; i<3-peaks.size(); i++) {
231 peaks.push_back(std::pair<size_t, size_t>(9999,1)); // stuff dummy
232 }
233 }
234 std::vector<size_t> result;
235 for(auto it=peaks.begin(); it != peaks.begin()+3; ++it) {
236 result.push_back((*it).first);
237 }
238
239 // sort peaks by index
240 std::sort(result.begin(), result.end(), [](const size_t &left, const size_t &right) { return left < right; });
241
242 return result;
243}

372size_t str2val(const std::string &hexstr) {
373 size_t res = 0;
374 if (hexstr.size()==0) return 0;
375 if (hexstr[0] == '$') {
376 for(auto it = hexstr.begin(); it != hexstr.end(); ++it) {
377 if (*it >='0' && *it <= '9') {
378 res = (res<<4) | ((*it) - '0');
379 } else if (*it>='a' && *it <= 'f') {
380 res = (res<<4) | ((*it) - 'a' + 10);
381 }
382 }
383 }
384 else if (hexstr[0] >= '0' && hexstr[0] <= '9') {
385 res = std::stoi(hexstr);
386 }
387 //std::cout << hexstr << " " << res << std::endl;
388 return res;
389}
