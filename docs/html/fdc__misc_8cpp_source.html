25std::vector<int> generate_interleaved_sector_list(size_t num_sector, int interleave = 1) {
26 std::vector<int> sector_list(num_sector, 0);
27 int pos = 0;
28 for (int i = 1; i <= num_sector; i++) {
29 while (sector_list[pos] != 0) {
30 if (++pos >= num_sector) pos = 0;
31 }
32 sector_list[pos] = i;
33 pos += interleave;
34 if (pos >= num_sector) pos = 0;
35 }
36 return sector_list;
37}

139std::vector<size_t> get_frequent_distribution(bit_array barray) {
140 barray.clear_wraparound_flag();
141 std::vector<size_t> freq_dist;
142 do {
143 size_t dist = barray.distance_to_next_pulse();
144 if(freq_dist.size() <= dist) {
145 freq_dist.resize(dist+1);
146 }
147 freq_dist[dist]++;
148 } while(!barray.is_wraparound());

152std::vector<size_t> find_peaks(const std::vector<size_t> &dist_freq) {
153 // LPF
154 std::vector<size_t> avg(dist_freq.size());
155 const int filt[5] { 1,2,3,2,1 };
156 const int flt_sum = 1+2+3+2+1;
157 for(int i = 0; i<dist_freq.size(); i++) {
158 size_t avg_tmp = 0;
159 for(int j=-2; j<=2; j++) {
160 int pos=0;
161 if (i+j < 0) {
162 pos= -(i+j); // data mirroring (lower boundary)
163 } else if(i+j >= dist_freq.size()) {
164 pos = 2*dist_freq.size() - (i+j) - 2; // data mirroring (upper boundary)
165 } else {
166 pos = i+j;
167 }
168 avg_tmp += dist_freq[pos] * filt[j+2];
169 }
170 avg[i] = avg_tmp / flt_sum;
171 }
172 //fdc_misc::display_histogram(avg); // for debug purpose
173
174 // detect peaks
175 std::vector<std::pair<size_t, size_t>> peaks;
176 bool flat_peak = false;
177 size_t lower_edge = 0;
178 for(size_t i=1; i<avg.size()-1-1; i++) {
179 if(avg[i-1] < avg[i] && avg[i+1] < avg[i]) { // a sharp peak
180 peaks.push_back(std::pair<size_t, size_t>(i, avg[i]));
181 flat_peak = false;
182 } else if(avg[i-1] < avg[i] && avg[i+1] == avg[i]) { // lower edge of a possible flat peak
183 flat_peak = true;
184 lower_edge = i;
185 } else if(avg[i-1] == avg[i] && avg[i+1] < avg[i] && flat_peak) { // upper edge of a flat peak
186 peaks.push_back(std::pair<size_t, size_t>((i+lower_edge)/2, avg[i]));
187 flat_peak = false;
188 } else if(avg[i-1] == avg[i] && avg[i+1] == avg[i] && flat_peak) { // top of flat peak
189 // nop
190 } else { // neither sharp peak, top of flat peak, nor edge of flat peak.
191 flat_peak = false;
192 }
193 }
194
195 // sort peaks by value (frequency)
196 std::sort(peaks.begin(), peaks.end(), [](const std::pair<size_t, size_t> &left, const std::pair<size_t, size_t> &right) { return left.second > right.second; });
197
198 if (peaks.size()<3) {
199 peaks.resize(3);
200 }
201 std::vector<size_t> result;
202 for(auto it=peaks.begin(); it != peaks.begin()+3; ++it) {
203 result.push_back((*it).first);
204 }
205
206 // sort peaks by index
207 std::sort(result.begin(), result.end(), [](const size_t &left, const size_t &right) { return left < right; });
208
209 return result;
210}

303size_t str2val(const std::string &hexstr) {
304 size_t res = 0;
305 if (hexstr.size()==0) return 0;
306 if (hexstr[0] == '$') {
307 for(auto it = hexstr.begin(); it != hexstr.end(); ++it) {
308 if (*it >='0' && *it <= '9') {
309 res = (res<<4) | ((*it) - '0');
310 } else if (*it>='a' && *it <= 'f') {
311 res = (res<<4) | ((*it) - 'a' + 10);
312 }
313 }
314 }
315 else if (hexstr[0] >= '0' && hexstr[0] <= '9') {
316 res = std::stoi(hexstr);
317 }
318 //std::cout << hexstr << " " << res << std::endl;
319 return res;
320}
