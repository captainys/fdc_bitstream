1// D88/D77 Disk format
2// Header
3// Ofst Size Contents Note
4// 0x00 17 Disk name (ended with '\0')
5// 0x11 9 Reserved
6// 0x1a 1 Write protect 0x00=No-protect 0x10=Protected
7// 0x1b 1 Disk type 0x00:2D, 0x10:2DD, 0x20:2HD, 0x30:1D, 0x40:1DD
8// 0x1c 4 Disk size Head + all tracks
9// 0x20 4*164 Track offset table Offset to the track data from the top of the file
10//
11// Sector
12// Ofst Size Contents Note
13// 0x00 1 Cylinder#
14// 0x01 1 Head#
15// 0x02 1 Sector#
16// 0x03 1 Secor size code (128<<N)
17// 0x04 2 # of sectors in the track
18// 0x06 1 Density 0x00:Double density, 0x40:Single density, 0x01:High density
19// 0x07 1 DAM 0x00:DAM, 0x10:DDAM
20// 0x08 1 Status 0x00:Normal, 0x10:DDAM, 0xa0:ID-CRC, 0xb0:DT-CRC, 0xe0:No IDAM, 0xf0:No DAM
21// 0x09 5 Reserved
22// 0x0e 2 Sector data size
23// 0x10 # Sector data

38void d77img::read(const std::string& file_name) {
39 byte_array image_data;
40 size_t image_size;
41
42 std::ifstream ifs;
43 ifs.open(file_name, std::ios_base::binary);
44 if (ifs.is_open() == false) {
45 std::cout << "Failed to open file" << std::endl;
46 return;
47 }
48 // Get image size
49 ifs.seekg(0, std::ios_base::end);
50 image_size = ifs.tellg();
51 ifs.seekg(0, std::ios_base::beg);
52
53 // Read image
54 image_data.resize(image_size);
55 ifs.read(reinterpret_cast<char*>(image_data.data()), image_size);
56
57 // Parse image
58 m_disk_name = image_data.get_string_z(0);
59 m_write_protect = image_data.get_byte_le(0x1a);
60 m_disk_type = image_data.get_byte_le(0x1b);
61 m_disk_size = image_data.get_dword_le(0x1c);
62
63 m_disk_data.clear();
64 m_disk_data.resize(164);
65
66 for (size_t track = 0; track < 164; track++) {
67 size_t track_offset = image_data.get_dword_le(0x20 + track * 4);
68 if (track_offset != 0) {
69 size_t num_sect = 0;
70 do {
71 sector_data sect;
72 sect.m_C = image_data.get_byte_le(track_offset + 0);
73 sect.m_H = image_data.get_byte_le(track_offset + 1);
74 sect.m_R = image_data.get_byte_le(track_offset + 2);
75 sect.m_N = image_data.get_byte_le(track_offset + 3);
76 sect.m_num_sectors = image_data.get_word_le(track_offset + 4);
77 sect.m_density = image_data.get_byte_le(track_offset + 6);
78 sect.m_dam_type = image_data.get_byte_le(track_offset + 7);
79 sect.m_status = image_data.get_byte_le(track_offset + 8);
80 sect.m_sector_data_length = image_data.get_word_le(track_offset + 0x0e);
81 sect.m_sector_data = image_data.get_block(track_offset + 0x10, (128 << (sect.m_N & 3)));
82 num_sect = sect.m_num_sectors;
83 m_disk_data[track].push_back(sect);
84
85 track_offset += 0x10 + sect.m_sector_data_length;
86
87 } while (m_disk_data[track].size() < num_sect);
88 }
89 }
90 ifs.close();
91}

93void d77img::write(std::string file_name) {
94 std::ofstream ofs;
95 byte_array image;
96
97 image.fill(0, 0x20 + 164 * 4, 0); // Header + track offset table (for 164 tracks)
98 image.set_string_z(0x00, m_disk_name, 17);
99 image.set_byte_le(0x1a, m_write_protect);
100 image.set_byte_le(0x1b, m_disk_type);
101 size_t track_offset = 0x20 + 164 * 4;
102 size_t num_track = media_max_track();
103
104 for (size_t track_n = 0; track_n < num_track; track_n++) {
105 if (track_n >= m_disk_data.size()) continue;
106 size_t num_sect = m_disk_data[track_n].size();
107 if (num_sect > 0) {
108 image.set_dword_le(0x20 + track_n * 4, track_offset); // track offset table
109
110 for (size_t sect_n = 0; sect_n < num_sect; sect_n++) {
111 sector_data sect = m_disk_data[track_n][sect_n];
112 image.set_byte_le(track_offset + 0x00, sect.m_C);
113 image.set_byte_le(track_offset + 0x01, sect.m_H);
114 image.set_byte_le(track_offset + 0x02, sect.m_R);
115 image.set_byte_le(track_offset + 0x03, sect.m_N);
116 image.set_word_le(track_offset + 0x04, num_sect);
117
118 image.set_byte_le(track_offset + 0x06, sect.m_density);
119 image.set_byte_le(track_offset + 0x07, sect.m_dam_type);
120 image.set_byte_le(track_offset + 0x08, sect.m_status);
121 image.set_word_le(track_offset + 0x0e, sect.m_sector_data_length);
122 for (size_t data_n = 0; data_n < sect.m_sector_data_length; data_n++) {
123 image.set_byte_le(track_offset + 0x10 + data_n, sect.m_sector_data[data_n]);
124 }
125 track_offset += 0x10 + sect.m_sector_data_length;
126 }
127 }
128 }
129 image.set_dword_le(0x1c, track_offset);
130
131 ofs.open(file_name, std::ios_base::binary);
132 if (ofs.is_open() == false) {
133 std::cout << "Failed to open file" << std::endl;
134 return;
135 }
136 ofs.write(reinterpret_cast<char*>(image.data()), image.size());
137}

139size_t d77img::media_max_track(void) {
140 size_t num_track;
141 switch (m_disk_type) {
142 case 0x00:
143 num_track = 84;
144 break;
145 case 0x10: // 2DD
146 case 0x20: // 2HD
147 num_track = 164;
148 break;
149 case 0x30: // 1D
150 num_track = 44;
151 break;
152 case 0x40: // 1DD
153 num_track = 84;
154 break;
155 default:
156 num_track = 84;
157 break;
158 }
159 return num_track;
160}

162byte_array d77img::read_sector(const size_t track, const size_t side, const size_t sect) {
163 if (track * 2 + side < media_max_track()) {
164 track_data& trk = m_disk_data[track * 2 + side];
165 for (size_t sect_n = 0; sect_n < trk.size(); sect_n++) {
166 if (trk[sect_n].m_C == track && trk[sect_n].m_R == sect) {
167 return trk[sect_n].m_sector_data;
168 }
169 }
170 }
171 return byte_array();
172}
