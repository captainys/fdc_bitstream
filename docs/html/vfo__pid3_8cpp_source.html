93double vfo_pid3::calc(double pulse_pos) {
94 //LPF
95 m_hist_ptr = (++m_hist_ptr) & (c_history_len-1);
96 m_pulse_pos_history[m_hist_ptr] = pulse_pos;
97 double sum = 0.f;
98 for(size_t i=1; i <= c_history_len; i++) {
99 sum += m_pulse_pos_history[ (m_hist_ptr+i) & (c_history_len-1) ] * i;
100 }
101 double avg = sum / m_coeff_sum;
102
103 // Phase error from the center of the bit cell
104 double phase_err_P = m_cell_center - avg;
105 double phase_err_D = phase_err_P - m_prev_phase_err;
106 m_phase_err_I += phase_err_P;
107 m_prev_phase_err = phase_err_P;
108
109 // Cell size adjustment == frequency correction + phase adjust
110 double new_cell_size = m_cell_size_ref
111 - (phase_err_P * m_phase_err_PC - phase_err_D * m_phase_err_DC + m_phase_err_I * m_phase_err_IC ) * m_current_gain; // Phase control (PID)
112
113 // FDD spindle variation (long term) = 2-2.5%
114 // FDD spindle wow/flutter variation (short term)= +-2-2.5%
115 // fast-slow combination = (2.5+2.5) *2 = 10%
116 // VFO oscillation center frequency variation by temp and power supply voltage variation = 5%
117 // Varition in total = 15%
118 constexpr double tolerance = 0.4f;
119 new_cell_size = limit(new_cell_size, m_cell_size_ref * (1.f/(1.f + tolerance)) , m_cell_size_ref * (1.0f + tolerance));
120
121 set_cell_size(new_cell_size);
122 m_prev_pulse_pos = pulse_pos;
123
124 return pulse_pos;
125}
