77void fdc_bitstream::write_track(const std::vector<uint8_t>& track_buf) {
78 uint16_t crc_val;
79 if (track_buf.size() == 0) return;
80 for (auto it = track_buf.begin(); it != track_buf.end(); ++it) {
81 uint8_t data = *it;
82 switch (data) {
83 case 0xf5: // marker A1
84 write_data(data, true);
85 m_crcgen.data(0xa1);
86 break;
87 case 0xf6: // marker C2
88 write_data(data, true);
89 m_crcgen.data(0xc2);
90 break;
91 case 0xf7: // write CRC (two bytes)
92 m_crcgen.data(0x00);
93 m_crcgen.data(0x00);
94 crc_val = m_crcgen.get();
95 write_data(crc_val >> 8);
96 write_data(crc_val & 0xffu);
97 break;
98 case 0xf8: // data address marks (reset CRC)
99 case 0xf9: // data address marks (reset CRC)
100 case 0xfa: // data address marks (reset CRC)
101 case 0xfb: // data address marks (reset CRC)
102 case 0xfe: // ID address marks (reset CRC)
103 m_crcgen.reset();
104 write_data(data);
105 m_crcgen.data(data);
106 break;
107 default:
108 write_data(data);
109 m_crcgen.data(data);
110 break;
111 }
112 }

124size_t fdc_bitstream::read_id(std::vector<uint8_t>& id_field, bool& crc_error) {
125 uint8_t read_data, mc_byte = 0;
126 bool missing_clock;
127 size_t read_count = 0;
128 id_field.clear();
129 crc_error = false;
130 size_t total_read_count = 0;
131 size_t read_start_pos = 0;
132 while (true) {
133 switch (m_state) {
134 case fdc_state::IDLE:
135 m_codec.mfm_read_byte(read_data, missing_clock, false, false);
136 if (missing_clock) {
137 m_state = fdc_state::CHECK_MARK;
138 mc_byte = read_data;
139 }
140 break;
141 case fdc_state::CHECK_MARK:
142 read_start_pos = m_codec.get_pos();
143 m_codec.mfm_read_byte(read_data, missing_clock, false, false);
144 if (missing_clock) {
145 mc_byte = read_data;
146 }
147 else if (mc_byte == 0xa1 && (read_data == 0xfc || read_data == 0xfd || read_data == 0xfe || read_data == 0xff /* IDAMs*/)) {
148 id_field.clear();
149 id_field.push_back(read_data);
150 m_crcgen.reset();
151 m_crcgen.data(read_data);
152 read_count = 4 + 2; // ID+CRC
153 m_state = fdc_state::READ_IDAM;
154 }
155 else {
156 m_state = fdc_state::IDLE;
157 }
158 break;
159 case fdc_state::READ_IDAM:
160 m_codec.mfm_read_byte(read_data, missing_clock, true, true);
161 id_field.push_back(read_data);
162 m_crcgen.data(read_data);
163 if (--read_count == 0) {
164 crc_error = ((m_crcgen.get() == 0) ? false : true);
165 id_field.erase(id_field.begin()); // remove the ID mark on the top
166 m_state = fdc_state::OPERATION_COMPLETED;
167 }
168 break;
169 default:
170 m_state = fdc_state::IDLE;
171 break;
172 }
173
174 if (m_state == fdc_state::OPERATION_COMPLETED) {
175 break;
176 }
177 total_read_count++;
178 if (total_read_count > m_codec.get_track_length()) {
179 break;
180 }
181 }
182 return read_start_pos;
183}

190std::vector<fdc_bitstream::id_field> fdc_bitstream::read_all_idam(void) {
191 std::vector<id_field> result;
192 id_field item;
193 memset(&item, 0, sizeof(id_field));
194 std::vector<uint8_t> sect_id;
195 bool crc_error = false;
196 clear_wraparound();
197 set_pos(0);
198 size_t pos;
199 do {
200 pos = read_id(sect_id, crc_error);
201 if (sect_id.size() > 0) {
202 item.C = sect_id[0];
203 item.H = sect_id[1];
204 item.R = sect_id[2];
205 item.N = sect_id[3];
206 item.crc_val = (sect_id[4] << 8) + sect_id[5];
207 item.crc_sts = crc_error;
208 item.pos = pos;
209 result.push_back(item);
210 }
211 } while (is_wraparound() == false);
212 return result;
213}

232 uint8_t read_data, mc_byte = 0;
233 bool missing_clock;
234 size_t read_count = 0;

239 size_t read_start_pos = 0;

242 case fdc_state::IDLE:
243 m_codec.mfm_read_byte(read_data, missing_clock, false, false);
244 if (missing_clock) {
245 m_state = fdc_state::CHECK_MARK;
246 mc_byte = read_data;
247 }
248 break;
249 case fdc_state::CHECK_MARK:

251 m_codec.mfm_read_byte(read_data, missing_clock, false, false);
252 if (missing_clock) {
253 mc_byte = read_data;
254 }
255 else if (mc_byte == 0xa1 && (read_data == 0xfa || read_data == 0xfb || read_data == 0xf8 || read_data == 0xf9)) {

260 dam_type = true;

264 m_crcgen.data(read_data);
265 read_count = sector_length_table[sect_length_code & 0b0011u] + 2; // sector size+CRC
266 m_state = fdc_state::READ_SECT;
267 }
268 else {
269 m_state = fdc_state::IDLE;
270 }
271 break;
272 case fdc_state::READ_SECT:
273 m_codec.mfm_read_byte(read_data, missing_clock, true, true);
274 m_crcgen.data(read_data);
275 sect_data.push_back(read_data);
276 if (--read_count == 0) {
277 crc_error = ((m_crcgen.get() == 0) ? false : true);
278 sect_data.pop_back(); // remove CRC field
279 sect_data.pop_back(); // remove CRC field
280 m_state = fdc_state::OPERATION_COMPLETED;
281 }
282 break;
283 default:
284 m_state = fdc_state::IDLE;
285 break;
286 }
287
288 if (m_state == fdc_state::OPERATION_COMPLETED) {
289 break;
290 }
291 total_read_count++;
292 if (total_read_count > m_codec.get_track_length()) {
293 break;
294 }
295 }
296 return read_start_pos;
297}

428fdc_bitstream::sector_data fdc_bitstream::read_sector(int trk, int sid, int sct) {
429 size_t index_hole_count = 0;
430 sector_data sect_data;
431 std::vector<uint8_t> sect_id;
432 std::vector<uint8_t> sect_body_data;
433 memset(&sect_data, 0, sizeof(sector_data));
434 bool crc_error = false;
435 bool dam_type = false;
436 size_t pos = 0;
437 if (trk > 43) trk = 43;
438 while (true) {
439 if (is_wraparound()) {
440 clear_wraparound();
441 if (++index_hole_count >= 4) { // FD179x/MB8876 needs to find the desired sector ID within 4 spins.
442 sect_data.record_not_found = true;
443 return sect_data;
444 }
445 }
446 pos = read_id(sect_id, crc_error);
447 sect_data.id_pos = pos;
448 if (sect_id[0] == trk && sect_id[2] == sct) { // FD179x/MB8876 won'nt compare 'head' field
449 if (crc_error == false) {
450 pos = read_sector_body(sect_id[3], sect_body_data, crc_error, dam_type);
451 sect_data.data_pos = pos;
452 size_t distance;
453 if (sect_data.data_pos >= sect_data.id_pos) {
454 distance = pos - sect_data.id_pos;
455 }
456 else {
457 distance = pos - sect_data.id_pos + get_track_length(); // Compensation for wrap around
458 }
459 if (distance > (43 + 9) * (4e6 / 500e3) * 8 * 2) { // DAM/DDAM must be found in 43 bytes (MFM, FDC179x/MB8876). +10 for IDAM field, *8 for 1 byte, *2 for (CLK+DATA).
460 sect_data.record_not_found = true;
461 }
462 else {
463 sect_data.data = sect_body_data;
464 sect_data.record_not_found = false;
465 return sect_data;
466 }
467 }
468 }
469 }

485bool fdc_bitstream::write_sector(int trk, int sid, int sct, bool dam_type, std::vector<uint8_t> &write_data, bool fluctuate) {
486 size_t index_hole_count = 0;
487 std::vector<uint8_t> sect_id;
488 bool crc_error = false;
489 if (trk > 43) trk = 43;
490 while (true) {
491 if (is_wraparound()) {
492 clear_wraparound();
493 if (++index_hole_count >= 4) { // FD179x/MB8876 needs to find the desired sector ID within 4 spins.
494 return false; // record not found error
495 }
496 }
497 read_id(sect_id, crc_error);
498 if (sect_id[0] == trk && sect_id[2] == sct) { // FD179x/MB8876 won'nt compare 'head' field
499 if (crc_error == false) {
500 if (fluctuate) {
501 // Fluctuate write start timing
502 size_t pos = get_pos();
503 pos += normal_distribution_random(); // +-3
504 if (pos < 0) pos = 0;
505 if (pos >= get_track_length()) pos = get_track_length() - 1;
506 set_pos(pos);
507 }
508 write_sector_body(write_data, dam_type);
509 return true;
510 }
