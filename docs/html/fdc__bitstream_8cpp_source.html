112void fdc_bitstream::write_track(const std::vector<uint8_t>& track_buf) {
113 uint16_t crc_val;
114 for (auto it = track_buf.begin(); it != track_buf.end(); ++it) {
115 uint8_t data = *it;
116 switch (data) {
117 case 0xf5: // marker A1
118 write_data(data, true);
119 m_crcgen.data(0xa1);
120 break;
121 case 0xf6: // marker C2
122 write_data(data, true);
123 m_crcgen.data(0xc2);
124 break;
125 case 0xf7: // write CRC (two bytes)
126 m_crcgen.data(0x00);
127 m_crcgen.data(0x00);
128 crc_val = m_crcgen.get();
129 write_data(crc_val >> 8);
130 write_data(crc_val & 0xffu);
131 break;
132 case 0xf8: // data address marks (reset CRC)
133 case 0xf9: // data address marks (reset CRC)
134 case 0xfa: // data address marks (reset CRC)
135 case 0xfb: // data address marks (reset CRC)
136 case 0xfe: // ID address marks (reset CRC)
137 m_crcgen.reset();
138 write_data(data);
139 m_crcgen.data(data);
140 break;
141 default:
142 write_data(data);
143 m_crcgen.data(data);
144 break;
145 }
146 }
147 m_codec.set_track_status_forcibly(true);
148}

159size_t fdc_bitstream::read_id(std::vector<uint8_t>& id_field, bool& crc_error) {
160 uint8_t read_data, mc_byte = 0;
161 bool missing_clock;
162 crc_error = false;
163 size_t read_count = 0;
164 id_field.clear();
165 if (m_codec.is_track_ready() == false) return 0;
166 size_t total_read_count = 0;
167 size_t read_start_pos = 0;
168 while (true) {
169 switch (m_state) {
170 case fdc_state::IDLE:
171 m_codec.mfm_read_byte(read_data, missing_clock, false, false);
172 if (missing_clock) {
173 m_state = fdc_state::CHECK_MARK;
174 mc_byte = read_data;
175 }
176 break;
177 case fdc_state::CHECK_MARK:
178 read_start_pos = m_codec.get_pos();
179 m_codec.mfm_read_byte(read_data, missing_clock, false, false);
180 if (missing_clock) {
181 mc_byte = read_data;
182 }
183 else if (mc_byte == 0xa1 && (read_data == 0xfc || read_data == 0xfd || read_data == 0xfe || read_data == 0xff /* IDAMs*/)) {
184 id_field.clear();
185 id_field.push_back(read_data);
186 m_crcgen.reset();
187 m_crcgen.data(read_data);
188 read_count = 4 + 2; // ID+CRC
189 m_state = fdc_state::READ_IDAM;
190 }
191 else {
192 m_state = fdc_state::IDLE;
193 }
194 break;
195 case fdc_state::READ_IDAM:
196 m_codec.mfm_read_byte(read_data, missing_clock, true, true);
197 id_field.push_back(read_data);
198 m_crcgen.data(read_data);
199 if (--read_count == 0) {
200 crc_error = ((m_crcgen.get() == 0) ? false : true);
201 id_field.erase(id_field.begin()); // remove the ID mark on the top
202 m_state = fdc_state::OPERATION_COMPLETED;
203 }
204 break;
205 default:
206 m_state = fdc_state::IDLE;
207 break;
208 }
209
210 if (m_state == fdc_state::OPERATION_COMPLETED) {
211 break;
212 }
213 total_read_count++;
214 if (total_read_count > m_codec.get_track_length()) {
215 break;
216 }
217 }
218 return read_start_pos;
219}

226std::vector<fdc_bitstream::id_field> fdc_bitstream::read_all_idam(void) {
227 std::vector<id_field> result;
228 id_field item;
229 memset(&item, 0, sizeof(id_field));
230 std::vector<uint8_t> sect_id;
231 bool crc_error = false;
232 if (m_codec.is_track_ready() == false) return result;
233 clear_wraparound();
234 set_pos(0);
235 while(true) {
236 sect_id.clear();
237 size_t pos = read_id(sect_id, crc_error);
238 if (is_wraparound() == true) break;
239 if (sect_id.size()>0) {
240 item.C = sect_id[0];
241 item.H = sect_id[1];
242 item.R = sect_id[2];
243 item.N = sect_id[3];
244 item.crc_val = (sect_id[4] << 8) + sect_id[5];
245 item.crc_sts = crc_error;
246 item.pos = pos;
247 result.push_back(item);
248 }
249 }
250 return result;
251}

270size_t fdc_bitstream::read_sector_body(size_t sect_length_code, std::vector<uint8_t>& sect_data, bool& crc_error, bool& dam_type, bool &record_not_found, bool timeout) {
271 std::vector<size_t> sector_length_table{ 128, 256, 512, 1024 };
272
273 sect_data.clear();
274 crc_error = false;
275 dam_type = false;
276 record_not_found = false;
277
278 size_t read_count = 0;
279 size_t total_read_count = 0;
280 size_t read_start_pos = 0;
281 int time_out_count;
282 time_out_count = timeout ? 43 : -1; // Time out (=record not found) in 43 bytes
283
284 uint8_t read_data, mc_byte = 0;
285 bool missing_clock;
286
287 while (true) {
288 switch (m_state) {
289 case fdc_state::IDLE:
290 m_codec.mfm_read_byte(read_data, missing_clock, false, false);
291 if (missing_clock) {
292 m_state = fdc_state::CHECK_MARK;
293 mc_byte = read_data;
294 }
295 break;
296 case fdc_state::CHECK_MARK:
297 read_start_pos = m_codec.get_pos();
298 m_codec.mfm_read_byte(read_data, missing_clock, false, false);
299 if (missing_clock) {
300 mc_byte = read_data;
301 }
302 else if (mc_byte == 0xa1 && (read_data == 0xfa || read_data == 0xfb || read_data == 0xf8 || read_data == 0xf9)) {
303 if (read_data == 0xfa || read_data == 0xfb) { // Normal DAM
304 dam_type = false;
305 }
306 else if (read_data == 0xf8 || read_data == 0xf9) { // DDAM
307 dam_type = true;
308 }
309 sect_data.clear();
310 m_crcgen.reset();
311 m_crcgen.data(read_data);
312 read_count = sector_length_table[sect_length_code & 0b0011u] + 2; // sector size+CRC
313 time_out_count = -1; // hereafter, no time out
314 m_state = fdc_state::READ_SECT;
315 }
316 else {
317 m_state = fdc_state::IDLE;
318 }
319 break;
320 case fdc_state::READ_SECT:
321 m_codec.mfm_read_byte(read_data, missing_clock, true, true);
322 m_crcgen.data(read_data);
323 sect_data.push_back(read_data);
324 if (--read_count == 0) {
325 crc_error = ((m_crcgen.get() == 0) ? false : true);
326 sect_data.pop_back(); // remove CRC field
327 sect_data.pop_back(); // remove CRC field
328 m_state = fdc_state::OPERATION_COMPLETED;
329 }
330 break;
331 default:
332 m_state = fdc_state::IDLE;
333 break;
334 }
335
336 if (m_state == fdc_state::OPERATION_COMPLETED) {
337 break;
338 }
339 if (time_out_count >0 ) {
340 if (--time_out_count == 0) { // DAM/DDAM didn't appear in 43 bytes
341 record_not_found = true;
342 return read_start_pos;
343 }
344 }
345 total_read_count++;
346 if (total_read_count > m_codec.get_track_length()) {
347 break;
348 }
349 }
350 return read_start_pos;
351}

365void fdc_bitstream::write_sector_body(std::vector<uint8_t> write_data, bool dam_type, bool write_crc) {
366 for (int i = 0; i < 2; i++) m_codec.mfm_write_byte(0x00, false, false); // WG==false == dummy write
367 for (int i = 0; i < 8; i++) m_codec.mfm_write_byte(0x00, false, false); // WG==false == dummy write
368 for (int i = 0; i < 1; i++) m_codec.mfm_write_byte(0x00, false, false); // WG==false == dummy write
369 for (int i = 0; i < 11; i++) m_codec.mfm_write_byte(0x00, false, false); // WG==false == dummy write
370 for (int i = 0; i < 12; i++) m_codec.mfm_write_byte(0x00); // 0x00 x 12
371 for (int i = 0; i < 3; i++) m_codec.mfm_write_byte(0xf5, true); // 0xf5 (0xa1*) x 3
372
373 // reset CRC
374 m_crcgen.reset();
375
376 // write DAM/DDAM
377 if (dam_type == false) {
378 m_codec.mfm_write_byte(0xfb); // 0xfb (normal DAM) x 1
379 m_crcgen.data(0xfb);
380 }
381 else {
382 m_codec.mfm_write_byte(0xf8); // 0xf8 (Deleted DAM) x 1
383 m_crcgen.data(0xf8);
384 }
385
386 // write sector data
387 for (auto it = write_data.begin(); it != write_data.end(); ++it) {
388 uint8_t data = *it;
389 m_codec.mfm_write_byte(data);
390 m_crcgen.data(data);
391 }
392
393 // write CRC
394 if (write_crc == true) {
395 m_crcgen.data(0x00);
396 m_crcgen.data(0x00);
397 uint16_t crc_val = m_crcgen.get();
398 m_codec.mfm_write_byte(crc_val >> 8);
399 m_codec.mfm_write_byte(crc_val & 0xffu);
400
401 m_codec.mfm_write_byte(0xff); // write 0xff (or 0x4f)
402 }
403}

498fdc_bitstream::sector_data fdc_bitstream::read_sector(int trk, int sct) {
499 size_t index_hole_count = 0;
500 sector_data sect_data;
501 std::vector<uint8_t> sect_id;
502 std::vector<uint8_t> sect_body_data;
503 memset(&sect_data, 0, sizeof(sector_data));
504 bool crc_error = false;
505 bool dam_type = false;
506 bool record_not_found = false;
507 while (true) {
508 if (is_wraparound()) {
509 clear_wraparound();
510 if (++index_hole_count >= 4) { // FD179x/MB8876 needs to find the desired sector ID within 4 spins.
511 sect_data.record_not_found = true;
512 return sect_data;
513 }
514 }
515 size_t pos = read_id(sect_id, crc_error);
516 sect_data.id_pos = pos;
517 if(sect_id.size()==0) continue; // Failed to read ID
518 if (sect_id[0] == trk && sect_id[2] == sct) { // FD179x/MB8876 won'nt compare 'head' field
519 if (crc_error == false) {
520 pos = read_sector_body(sect_id[3], sect_body_data, crc_error, dam_type, record_not_found);
521 sect_data.data_pos = pos;
522 size_t distance;
523 if (sect_data.data_pos >= sect_data.id_pos) {
524 distance = pos - sect_data.id_pos;
525 }
526 else {
527 distance = pos - sect_data.id_pos + get_track_length(); // Compensation for wrap around
528 }
529 if (distance > (43 + 9) * (4e6 / 500e3) * 8 * 2) { // DAM/DDAM must be found in 43 bytes (MFM, FDC179x/MB8876). +10 for IDAM field, *8 for 1 byte, *2 for (CLK+DATA).
530 sect_data.record_not_found = true;
531 }
532 else {
533 sect_data.data = sect_body_data;
534 sect_data.record_not_found = record_not_found;
535 sect_data.crc_sts = crc_error;
536 return sect_data;
537 }
538 }
539 }

555bool fdc_bitstream::write_sector(int trk, int sct, bool dam_type, std::vector<uint8_t> &write_data, bool fluctuate) {
556 size_t index_hole_count = 0;
557 std::vector<uint8_t> sect_id;
558 bool crc_error = false;
559 while (true) {
560 if (is_wraparound()) {
561 clear_wraparound();
562 if (++index_hole_count >= 4) { // FD179x/MB8876 needs to find the desired sector ID within 4 spins.
563 return false; // record not found error
564 }
565 }
566 read_id(sect_id, crc_error);
567 if(sect_id.size() == 0) continue;
568 if (sect_id[0] == trk && sect_id[2] == sct) { // FD179x/MB8876 won'nt compare 'head' field
569 if (crc_error == false) {
570 if (fluctuate) {
571 // Fluctuate write start timing
572 int64_t pos = static_cast<int64_t>(get_pos());
573 pos += normal_distribution_random(); // +-3
574 if (pos < 0) pos = 0;
575 if (pos >= get_track_length()) pos = static_cast<int64_t>(get_track_length()) - 1;
576 set_pos(static_cast<size_t>(pos));
577 }
578 write_sector_body(write_data, dam_type);
579 return true;
580 }
