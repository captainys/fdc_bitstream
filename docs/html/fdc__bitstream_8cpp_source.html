80void fdc_bitstream::write_track(const std::vector<uint8_t>& track_buf) {
81 uint16_t crc_val;
82 for (auto it = track_buf.begin(); it != track_buf.end(); ++it) {
83 uint8_t data = *it;
84 switch (data) {
85 case 0xf5: // marker A1
86 write_data(data, true);
87 m_crcgen.data(0xa1);
88 break;
89 case 0xf6: // marker C2
90 write_data(data, true);
91 m_crcgen.data(0xc2);
92 break;
93 case 0xf7: // write CRC (two bytes)
94 m_crcgen.data(0x00);
95 m_crcgen.data(0x00);
96 crc_val = m_crcgen.get();
97 write_data(crc_val >> 8);
98 write_data(crc_val & 0xffu);
99 break;
100 case 0xf8: // data address marks (reset CRC)
101 case 0xf9: // data address marks (reset CRC)
102 case 0xfa: // data address marks (reset CRC)
103 case 0xfb: // data address marks (reset CRC)
104 case 0xfe: // ID address marks (reset CRC)
105 m_crcgen.reset();
106 write_data(data);
107 m_crcgen.data(data);
108 break;
109 default:
110 write_data(data);
111 m_crcgen.data(data);
112 break;
113 }
114 }
115 m_codec.set_track_status_forcibly(true);
116}

127size_t fdc_bitstream::read_id(std::vector<uint8_t>& id_field, bool& crc_error) {
128 uint8_t read_data, mc_byte = 0;
129 bool missing_clock;
130 crc_error = false;
131 size_t read_count = 0;
132 id_field.clear();
133 if (m_codec.is_track_ready() == false) return 0;
134 size_t total_read_count = 0;
135 size_t read_start_pos = 0;
136 while (true) {
137 switch (m_state) {
138 case fdc_state::IDLE:
139 m_codec.mfm_read_byte(read_data, missing_clock, false, false);
140 if (missing_clock) {
141 m_state = fdc_state::CHECK_MARK;
142 mc_byte = read_data;
143 }
144 break;
145 case fdc_state::CHECK_MARK:
146 read_start_pos = m_codec.get_pos();
147 m_codec.mfm_read_byte(read_data, missing_clock, false, false);
148 if (missing_clock) {
149 mc_byte = read_data;
150 }
151 else if (mc_byte == 0xa1 && (read_data == 0xfc || read_data == 0xfd || read_data == 0xfe || read_data == 0xff /* IDAMs*/)) {
152 id_field.clear();
153 id_field.push_back(read_data);
154 m_crcgen.reset();
155 m_crcgen.data(read_data);
156 read_count = 4 + 2; // ID+CRC
157 m_state = fdc_state::READ_IDAM;
158 }
159 else {
160 m_state = fdc_state::IDLE;
161 }
162 break;
163 case fdc_state::READ_IDAM:
164 m_codec.mfm_read_byte(read_data, missing_clock, true, true);
165 id_field.push_back(read_data);
166 m_crcgen.data(read_data);
167 if (--read_count == 0) {
168 crc_error = ((m_crcgen.get() == 0) ? false : true);
169 id_field.erase(id_field.begin()); // remove the ID mark on the top
170 m_state = fdc_state::OPERATION_COMPLETED;
171 }
172 break;
173 default:
174 m_state = fdc_state::IDLE;
175 break;
176 }
177
178 if (m_state == fdc_state::OPERATION_COMPLETED) {
179 break;
180 }
181 total_read_count++;
182 if (total_read_count > m_codec.get_track_length()) {
183 break;
184 }
185 }
186 return read_start_pos;
187}

194std::vector<fdc_bitstream::id_field> fdc_bitstream::read_all_idam(void) {
195 std::vector<id_field> result;
196 id_field item;
197 memset(&item, 0, sizeof(id_field));
198 std::vector<uint8_t> sect_id;
199 bool crc_error = false;
200 if (m_codec.is_track_ready() == false) return result;
201 clear_wraparound();
202 set_pos(0);
203 while(true) {
204 size_t pos = read_id(sect_id, crc_error);
205 if (is_wraparound() == true) break;
206 if (sect_id.size() ==6) {
207 item.C = sect_id[0];
208 item.H = sect_id[1];
209 item.R = sect_id[2];
210 item.N = sect_id[3];
211 item.crc_val = (sect_id[4] << 8) + sect_id[5];
212 item.crc_sts = crc_error;
213 item.pos = pos;
214 result.push_back(item);
215 }
216 }
217 return result;
218}

235size_t fdc_bitstream::read_sector_body(size_t sect_length_code, std::vector<uint8_t>& sect_data, bool& crc_error, bool& dam_type) {
236 std::vector<size_t> sector_length_table{ 128, 256, 512, 1024 };
237 uint8_t read_data, mc_byte = 0;
238 bool missing_clock;
239 size_t read_count = 0;
240 sect_data.clear();
241 crc_error = false;
242 dam_type = false;
243 size_t total_read_count = 0;
244 size_t read_start_pos = 0;
245 while (true) {
246 switch (m_state) {
247 case fdc_state::IDLE:
248 m_codec.mfm_read_byte(read_data, missing_clock, false, false);
249 if (missing_clock) {
250 m_state = fdc_state::CHECK_MARK;
251 mc_byte = read_data;
252 }
253 break;
254 case fdc_state::CHECK_MARK:
255 read_start_pos = m_codec.get_pos();
256 m_codec.mfm_read_byte(read_data, missing_clock, false, false);
257 if (missing_clock) {
258 mc_byte = read_data;
259 }
260 else if (mc_byte == 0xa1 && (read_data == 0xfa || read_data == 0xfb || read_data == 0xf8 || read_data == 0xf9)) {
261 if (read_data == 0xfa || read_data == 0xfb) { // Normal DAM
262 dam_type = false;
263 }
264 else if (read_data == 0xf8 || read_data == 0xf9) { // DDAM
265 dam_type = true;
266 }
267 sect_data.clear();
268 m_crcgen.reset();
269 m_crcgen.data(read_data);
270 read_count = sector_length_table[sect_length_code & 0b0011u] + 2; // sector size+CRC
271 m_state = fdc_state::READ_SECT;
272 }
273 else {
274 m_state = fdc_state::IDLE;
275 }
276 break;
277 case fdc_state::READ_SECT:
278 m_codec.mfm_read_byte(read_data, missing_clock, true, true);
279 m_crcgen.data(read_data);
280 sect_data.push_back(read_data);
281 if (--read_count == 0) {
282 crc_error = ((m_crcgen.get() == 0) ? false : true);
283 sect_data.pop_back(); // remove CRC field
284 sect_data.pop_back(); // remove CRC field
285 m_state = fdc_state::OPERATION_COMPLETED;
286 }
287 break;
288 default:
289 m_state = fdc_state::IDLE;
290 break;
291 }
292
293 if (m_state == fdc_state::OPERATION_COMPLETED) {
294 break;
295 }
296 total_read_count++;
297 if (total_read_count > m_codec.get_track_length()) {
298 break;
299 }
300 }
301 return read_start_pos;
302}

316void fdc_bitstream::write_sector_body(std::vector<uint8_t> write_data, bool dam_type, bool write_crc) {
317 for (int i = 0; i < 2; i++) m_codec.mfm_write_byte(0x00, false, false); // WG==false == dummy write
318 for (int i = 0; i < 8; i++) m_codec.mfm_write_byte(0x00, false, false); // WG==false == dummy write
319 for (int i = 0; i < 1; i++) m_codec.mfm_write_byte(0x00, false, false); // WG==false == dummy write
320 for (int i = 0; i < 11; i++) m_codec.mfm_write_byte(0x00, false, false); // WG==false == dummy write
321 for (int i = 0; i < 12; i++) m_codec.mfm_write_byte(0x00); // 0x00 x 12
322 for (int i = 0; i < 3; i++) m_codec.mfm_write_byte(0xf5, true); // 0xf5 (0xa1*) x 3
323
324 // reset CRC
325 m_crcgen.reset();
326
327 // write DAM/DDAM
328 if (dam_type == false) {
329 m_codec.mfm_write_byte(0xfb); // 0xfb (normal DAM) x 1
330 m_crcgen.data(0xfb);
331 }
332 else {
333 m_codec.mfm_write_byte(0xf8); // 0xf8 (Deleted DAM) x 1
334 m_crcgen.data(0xf8);
335 }
336
337 // write sector data
338 for (auto it = write_data.begin(); it != write_data.end(); ++it) {
339 uint8_t data = *it;
340 m_codec.mfm_write_byte(data);
341 m_crcgen.data(data);
342 }
343
344 // write CRC
345 if (write_crc == true) {
346 m_crcgen.data(0x00);
347 m_crcgen.data(0x00);
348 uint16_t crc_val = m_crcgen.get();
349 m_codec.mfm_write_byte(crc_val >> 8);
350 m_codec.mfm_write_byte(crc_val & 0xffu);
351
352 m_codec.mfm_write_byte(0xff); // write 0xff (or 0x4f)
353 }

442fdc_bitstream::sector_data fdc_bitstream::read_sector(int trk, int sid, int sct) {
443 size_t index_hole_count = 0;
444 sector_data sect_data;
445 std::vector<uint8_t> sect_id;
446 std::vector<uint8_t> sect_body_data;
447 memset(&sect_data, 0, sizeof(sector_data));
448 bool crc_error = false;
449 bool dam_type = false;
450 if (trk > 43) trk = 43;
451 while (true) {
452 if (is_wraparound()) {
453 clear_wraparound();
454 if (++index_hole_count >= 4) { // FD179x/MB8876 needs to find the desired sector ID within 4 spins.
455 sect_data.record_not_found = true;
456 return sect_data;
457 }
458 }
459 size_t pos = read_id(sect_id, crc_error);
460 sect_data.id_pos = pos;
461 if (sect_id[0] == trk && sect_id[2] == sct) { // FD179x/MB8876 won'nt compare 'head' field
462 if (crc_error == false) {
463 pos = read_sector_body(sect_id[3], sect_body_data, crc_error, dam_type);
464 sect_data.data_pos = pos;
465 size_t distance;
466 if (sect_data.data_pos >= sect_data.id_pos) {
467 distance = pos - sect_data.id_pos;
468 }
469 else {
470 distance = pos - sect_data.id_pos + get_track_length(); // Compensation for wrap around
471 }
472 if (distance > (43 + 9) * (4e6 / 500e3) * 8 * 2) { // DAM/DDAM must be found in 43 bytes (MFM, FDC179x/MB8876). +10 for IDAM field, *8 for 1 byte, *2 for (CLK+DATA).
473 sect_data.record_not_found = true;
474 }
475 else {
476 sect_data.data = sect_body_data;
477 sect_data.record_not_found = false;
478 return sect_data;
479 }
480 }
481 }
482 }

498bool fdc_bitstream::write_sector(int trk, int sid, int sct, bool dam_type, std::vector<uint8_t> &write_data, bool fluctuate) {
499 size_t index_hole_count = 0;
500 std::vector<uint8_t> sect_id;
501 bool crc_error = false;
502 if (trk > 43) trk = 43;
503 while (true) {
504 if (is_wraparound()) {
505 clear_wraparound();
506 if (++index_hole_count >= 4) { // FD179x/MB8876 needs to find the desired sector ID within 4 spins.
507 return false; // record not found error
508 }
509 }
510 read_id(sect_id, crc_error);
511 if (sect_id[0] == trk && sect_id[2] == sct) { // FD179x/MB8876 won'nt compare 'head' field
512 if (crc_error == false) {
513 if (fluctuate) {
514 // Fluctuate write start timing
515 int64_t pos = static_cast<int64_t>(get_pos());
516 pos += normal_distribution_random(); // +-3
517 if (pos < 0) pos = 0;
518 if (pos >= get_track_length()) pos = static_cast<int64_t>(get_track_length()) - 1;
519 set_pos(static_cast<size_t>(pos));
520 }
521 write_sector_body(write_data, dam_type);
522 return true;
523 }
