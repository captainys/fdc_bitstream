42double vfo_pid::calc(double pulse_pos) {
43 // compare the position of the current and previous pulses in the bit cell (phase_diff -=slow freq, +=fast freq)
44 double phase_diff = m_prev_pulse_pos - pulse_pos;
45 // phase correction. assuming phase difference is less than 180deg.
46 if(phase_diff > m_cell_size / 2.f) {
47 phase_diff = phase_diff - m_cell_size;
48 } else if (phase_diff < -(m_cell_size / 2.f)) {
49 phase_diff = phase_diff + m_cell_size;
50 }
51
52 // Phase error from the center of the bit cell
53 double phase_err = m_cell_center - pulse_pos;
54 double phase_err_diff = phase_err - m_prev_phase_err;
55 m_phase_err_integral += phase_err;
56 m_prev_phase_err = phase_err;
57
58 // Cell size adjustment == frequency correction
59 m_phase_diff_integral += phase_diff;
60 double new_cell_size = m_cell_size_ref - (m_phase_diff_integral * (1.0f/24.f) ) * m_current_gain -
61 (phase_err * (1.0f/2.f) - phase_err_diff * (1.0f/8.f) + m_phase_err_integral * (1.0f/128.f));
62 constexpr double range = 0.02f;
63 new_cell_size = limit(new_cell_size, m_cell_size_ref * (1-range), m_cell_size_ref * (1+range));
64 m_prev_pulse_pos = pulse_pos;
65
66 set_cell_size(new_cell_size);
67
68 return pulse_pos;
