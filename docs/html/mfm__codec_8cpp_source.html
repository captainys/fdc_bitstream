149// How data separator work:
150//
151// | WWWW | WWWW | WWWW | WWWW | WWWW | WWWW | WWWW | WWWW | <- Data window(W), Data cell boundary(|)
152// P P P P P <- Data pulses
153// 1 0 1 0 0 1 0 0 <- Data reading (only the data pulses within the data window will be considered)
154//
155// 01234567
156// | WWWW | <= In this case, bit cell size = 8, data window ofst = 2, data window size = 4
157//

165int mfm_codec::read_bit_ds(void) {
166 int bit_reading = 0;
167 double cell_center;
168 size_t loop_count = 0; // for timeout check
169
170 if (is_track_ready() == false) {
171 return -1;
172 }
173
174 do {
175 // check if the next bit is within the next data window
176 if (m_distance_to_next_pulse < m_bit_cell_size) {
177 // check pulse position
178 if (m_distance_to_next_pulse >= m_data_window_ofst &&
179 // regular pulse (within the data window)
180 m_distance_to_next_pulse < m_data_window_ofst + m_data_window_size) {
181 bit_reading = 1;
182 }
183 else {
184 // irregular pulse
185#ifdef DEBUG
186 std::cout << '?';
187#endif
188 }
189 // Adjust pulse phase (imitate PLL/VFO operation)
190 // Limit the PLL/VFO operation frequency and introduce fluctuation with the random generator (certain fluctuation is required to reproduce some copy protection)
191 if (m_fluctuation == false || m_rnd() % m_fluctuator_denominator >= m_fluctuator_numerator) {
192 cell_center = m_data_window_ofst + m_data_window_size / 2.f;
193 double error = m_distance_to_next_pulse - cell_center;
194
195 // Data pulse position adjustment == phase correction
196 m_distance_to_next_pulse -= error * 0.05f * m_vfo_gain;
197
198 // Cell size adjustment == frequency correction
199 double new_cell_size = m_bit_cell_size + error * 0.01f * m_vfo_gain;
200 // Limit the range of cell size
201 if (new_cell_size < m_bit_cell_size_ref * 0.8) new_cell_size = m_bit_cell_size_ref * 0.8;
202 if (new_cell_size > m_bit_cell_size_ref * 1.2) new_cell_size = m_bit_cell_size_ref * 1.2;
203 set_cell_size(new_cell_size);
204 }
205 size_t distance = m_track.distance_to_next_bit1();
206#if 0
207 // give timing fluctuation (intentionally - to immitate the spndle rotation fluctuation)
208 size_t rand_num = m_rnd() % 128;
209 if (rand_num == 0) {
210 distance++;
211 }
212 else if (rand_num == 1) {
213 distance--;
214 }
215#endif
216 m_distance_to_next_pulse += distance;
217
218 if (loop_count++ > m_bit_cell_size) { // time out
219 return -1;
220 }
221 }
222 } while (m_distance_to_next_pulse < m_bit_cell_size);
223 // advance bit cell position
224 if (m_distance_to_next_pulse >= m_bit_cell_size) {
225 m_distance_to_next_pulse -= m_bit_cell_size;
226 }
227 return bit_reading;
228}

255bool mfm_codec::mfm_read_byte(uint8_t& data, bool& missing_clock, bool ignore_missing_clock, bool ignore_sync_field) {
256 size_t decode_count = 0;
257 missing_clock = false;
258
259 if (is_track_ready() == false) {
260 data = 0;
261 missing_clock = false;
262 return false;
263 }
264
265 do {
266 int bit_data = read_bit_ds();
267 if (bit_data == -1) {
268 data = 0;
269 missing_clock = false;
270 return false;
271 }
272 decode_count++;
273 if (m_track.is_wraparound()) {
274 m_track.clear_wraparound_flag();
275 m_wraparound = true;
276 }
277 m_bit_stream = (m_bit_stream << 1) | bit_data;
278
279 if (ignore_missing_clock == false) {
280 if ((m_bit_stream & 0x0ffffu) == m_missing_clock_a1) { // Missing clock 0xA1 pattern
281 data = 0xa1;
282 missing_clock = true;
283 return true;
284 }
285 if ((m_bit_stream & 0x0ffffu) == m_missing_clock_c2) { // Missing clock 0xC2 pattern
286 data = 0xc2;
287 missing_clock = true;
288 return true;
289 }
290 }
291 if (ignore_sync_field == false) {
292 if (m_bit_stream == m_pattern_00) {
293 decode_count &= ~0b01u; // C/D synchronize
294 set_gain(m_vfo_gain_h);
295 }
296 else {
297 set_gain(m_vfo_gain_l);
298 }
299 }
300 } while (decode_count < 16);
301
302 // Extract only 'D' bits (exclude 'C' bits). MFM data is 'CDCDCDCD..CD'.
303 uint8_t read_data = 0;
304 for (uint16_t bit_pos = 0x4000; bit_pos > 0; bit_pos >>= 2) {
305 read_data = (read_data << 1) | (m_bit_stream & bit_pos ? 1 : 0);
306 }
307 data = read_data;
308 missing_clock = false;
309 return true;
310}

320uint16_t mfm_codec::mfm_encoder(uint8_t data, bool mode) {
321 // Data swap for special code
322 uint8_t write_data = data;
323 if (mode == true) { // special coding mode (F5 and F6 will be converted to A1* C2* with missing-clock pattern respectively)
324 switch (data) {
325 case 0xf5:
326 write_data = 0xa1;
327 break;
328 case 0xf6:
329 write_data = 0xc2;
330 break;
331 }
332 }
333 // MFM encoding
334 uint16_t bit_pattern = 0;
335 uint16_t mfm_bit_pattern = 0;
336 int current_bit;
337 int clock_bit;
338 for (size_t bit_pos = 0x80; bit_pos != 0; bit_pos >>= 1) {
339 current_bit = (write_data & bit_pos) ? 1 : 0;
340 clock_bit = (m_prev_write_bit == 0 && current_bit == 0) ? 1 : 0;
341 mfm_bit_pattern = (mfm_bit_pattern << 1) | clock_bit; // clock
342 mfm_bit_pattern = (mfm_bit_pattern << 1) | current_bit; // data
343 m_prev_write_bit = current_bit;
344 }
345 // Missing clock operation
346 if (mode == true) {
347 switch (data) {
348 case 0xf5:
349 mfm_bit_pattern &= ~0b100000u; // remove a clock bit
350 break;
351 case 0xf6:
352 mfm_bit_pattern &= ~0b10000000u; // remove a clock bit
353 break;
354 }
355 }
356 return mfm_bit_pattern;
357}

367void mfm_codec::mfm_write_byte(uint8_t data, bool mode, bool write_gate) {
368 uint16_t bit_pattern = mfm_encoder(data, mode);
369 for (uint16_t bit_pos = 0x8000; bit_pos != 0; bit_pos >>= 1) {
370 int bit = bit_pattern & bit_pos ? 1 : 0;
371 if (write_gate == true) {
372 for (size_t i = 0; i < m_bit_width; i++) {
373 if (m_bit_width / 2 == i) m_track.write_stream(bit); // write data pulse at the center of the bit cell
374 else m_track.write_stream(0);
375 }
376 }
377 else {
378 for (size_t i = 0; i < m_bit_width; i++) {
379 m_track.advance_stream_pos(); // advance stream pointer without actual data write (dummy write)
380 }
381 }
382 }
