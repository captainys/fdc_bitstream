157// How data separator work:
158//
159// | WWWW | WWWW | WWWW | WWWW | WWWW | WWWW | WWWW | WWWW | <- Data window(W), Data cell boundary(|)
160// P P P P P <- Data pulses
161// 1 0 1 0 0 1 0 0 <- Data reading (only the data pulses within the data window will be considered)
162//
163// 01234567
164// | WWWW | <= In this case, bit cell size = 8, data window ofst = 2, data window size = 4
165//

173int mfm_codec::read_bit_ds(void) {
174 int bit_reading = 0;
175 double cell_center;
176 size_t loop_count = 0; // for timeout check
177
178 if (is_track_ready() == false) {
179 return -1;
180 }
181
182 do {
183 // check if the next bit is within the next data window
184 if (m_distance_to_next_pulse < m_vfo->m_cell_size) {
185 // check pulse position
186 if (m_distance_to_next_pulse >= m_vfo->m_window_ofst &&
187 // regular pulse (within the data window)
188 m_distance_to_next_pulse < m_vfo->m_window_ofst + m_vfo->m_window_size) {
189 bit_reading = 1;
190 }
191 else {
192 // irregular pulse
193#ifdef DEBUG
194 std::cout << '?';
195#endif
196 }
197 // Adjust pulse phase (imitate PLL/VFO operation)
198 // Limit the PLL/VFO operation frequency and introduce fluctuation with the random generator (certain fluctuation is required to reproduce some copy protection)
199 if ((static_cast<double>(m_rnd()) / static_cast<double>(INT32_MAX)) >= m_vfo_suspension_rate) {
200 m_distance_to_next_pulse = m_vfo->calc(m_distance_to_next_pulse);
201 }
202 size_t distance = m_track.distance_to_next_bit1();
203#if 0
204 // give timing fluctuation (intentionally - to immitate the spndle rotation fluctuation)
205 size_t rand_num = m_rnd() % 128;
206 if (rand_num == 0) {
207 distance++;
208 }
209 else if (rand_num == 1) {
210 distance--;
211 }
212#endif
213 m_distance_to_next_pulse += distance;
214
215 if (loop_count++ > m_vfo->m_cell_size) { // time out
216 return -1;
217 }
218 }
219 } while (m_distance_to_next_pulse < m_vfo->m_cell_size);
220 // advance bit cell position
221 if (m_distance_to_next_pulse >= m_vfo->m_cell_size) {
222 m_distance_to_next_pulse -= m_vfo->m_cell_size;
223 }
224 return bit_reading;
225}

258bool mfm_codec::mfm_read_byte(uint8_t& data, bool& missing_clock, bool ignore_missing_clock, bool ignore_sync_field) {
259 size_t decode_count = 0;
260 missing_clock = false;
261
262 if (is_track_ready() == false) {
263 data = 0;
264 missing_clock = false;
265 return false;
266 }
267
268 do {
269 int bit_data = read_bit_ds();
270 if (bit_data == -1) {
271 data = 0;
272 missing_clock = false;
273 return false;
274 }
275 decode_count++;
276 if (m_track.is_wraparound()) {
277 m_track.clear_wraparound_flag();
278 m_wraparound = true;
279 }
280 m_bit_stream = (m_bit_stream << 1) | bit_data;
281
282 if (ignore_missing_clock == false) {
283 if ((m_bit_stream & 0x0ffffu) == m_missing_clock_a1) { // Missing clock 0xA1 pattern
284 data = 0xa1;
285 missing_clock = true;
286 return true;
287 }
288 if ((m_bit_stream & 0x0ffffu) == m_missing_clock_c2) { // Missing clock 0xC2 pattern
289 data = 0xc2;
290 missing_clock = true;
291 return true;
292 }
293 }
294 if (ignore_sync_field == false) {
295 if (m_bit_stream == m_pattern_00) {
296 decode_count &= ~0b01u; // C/D synchronize
297 set_vfo_gain(mfm_codec::gain_state::high);
298 }
299 else {
300 set_vfo_gain(mfm_codec::gain_state::low);
301 }
302 }
303 } while (decode_count < 16);
304
305 // Extract only 'D' bits (exclude 'C' bits). MFM data is 'CDCDCDCD..CD'.
306 uint8_t read_data = 0;
307 for (uint16_t bit_pos = 0x4000; bit_pos > 0; bit_pos >>= 2) {
308 read_data = (read_data << 1) | ((m_bit_stream & bit_pos) ? 1 : 0);
309 }
310 data = read_data;
311 missing_clock = false;
312 return true;
313}

323uint16_t mfm_codec::mfm_encoder(uint8_t data, bool mode) {
324 // Data swap for special code
325 uint8_t write_data = data;
326 if (mode == true) { // special coding mode (F5 and F6 will be converted to A1* C2* with missing-clock pattern respectively)
327 switch (data) {
328 case 0xf5:
329 write_data = 0xa1;
330 break;
331 case 0xf6:
332 write_data = 0xc2;
333 break;
334 }
335 }
336 // MFM encoding
337 uint16_t mfm_bit_pattern = 0;
338 for (size_t bit_pos = 0x80; bit_pos != 0; bit_pos >>= 1) {
339 int current_bit = (write_data & bit_pos) ? 1 : 0;
340 int clock_bit = (m_prev_write_bit == 0 && current_bit == 0) ? 1 : 0;
341 mfm_bit_pattern = (mfm_bit_pattern << 1) | clock_bit; // clock
342 mfm_bit_pattern = (mfm_bit_pattern << 1) | current_bit; // data
343 m_prev_write_bit = current_bit;
344 }
345 // Missing clock operation
346 if (mode == true) {
347 switch (data) {
348 case 0xf5:
349 mfm_bit_pattern &= ~0b100000u; // remove a clock bit
350 break;
351 case 0xf6:
352 mfm_bit_pattern &= ~0b10000000u; // remove a clock bit
353 break;
354 }
355 }
356 return mfm_bit_pattern;
357}

367void mfm_codec::mfm_write_byte(uint8_t data, bool mode, bool write_gate) {
368 uint16_t bit_pattern = mfm_encoder(data, mode);
369 for (uint16_t bit_pos = 0x8000; bit_pos != 0; bit_pos >>= 1) {
370 int bit = (bit_pattern & bit_pos) ? 1 : 0;
371 if (write_gate == true) {
372 for (size_t i = 0; i < m_bit_width_w; i++) {
373 if (m_bit_width_w / 2 == i) m_track.write_stream(bit); // write data pulse at the center of the bit cell
374 else m_track.write_stream(0);
375 }
376 }
377 else {
378 for (size_t i = 0; i < m_bit_width_w; i++) {
379 m_track.advance_stream_pos(); // advance stream pointer without actual data write (dummy write)
380 }
381 }
382 }
