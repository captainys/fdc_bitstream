167// How data separator work:
168//
169// | WWWW | WWWW | WWWW | WWWW | WWWW | WWWW | WWWW | WWWW | <- Data window(W), Data cell boundary(|)
170// P P P P P <- Data pulses
171// 1 0 1 0 0 1 0 0 <- Data reading (only the data pulses within the data window will be considered)
172//
173// 01234567
174// | WWWW | <= In this case, bit cell size = 8, data window ofst = 2, data window size = 4
175//

183int mfm_codec::read_bit_ds(void) {
184 int bit_reading = 0;
185 size_t loop_count = 0; // for timeout check
186
187 if (is_track_ready() == false) {
188 return -1;
189 }
190
191 do {
192 // check if the next bit is within the next data window
193 if (m_distance_to_next_pulse < m_vfo->m_cell_size) {
194 // check pulse position
195 if (m_distance_to_next_pulse >= m_vfo->m_window_ofst &&
196 m_distance_to_next_pulse < m_vfo->m_window_ofst + m_vfo->m_window_size) {
197 bit_reading = 1; // regular pulse (within the data window)
198
199 }
200 else {
201#ifdef DEBUG
202 std::cout << '?'; // irregular pulse
203#endif
204 }
205 // Adjust pulse phase (imitate PLL/VFO operation)
206 // Limit the PLL/VFO operation frequency and introduce fluctuation with the random generator (certain fluctuation is required to reproduce some copy protection)
207 if ((static_cast<double>(m_rnd()) / static_cast<double>(INT32_MAX)) >= m_vfo_suspension_rate) {
208 m_distance_to_next_pulse = m_vfo->calc(m_distance_to_next_pulse);
209 }
210
211 size_t distance = m_track.distance_to_next_pulse();
212#if 0
213 // give timing fluctuation (intentionally - to immitate the spndle rotation fluctuation)
214 size_t rand_num = m_rnd() % 128;
215 if (rand_num == 0) {
216 distance++;
217 }
218 else if (rand_num == 1) {
219 distance--;
220 }
221#endif
222 m_distance_to_next_pulse += distance;
223
224 if (loop_count++ > m_vfo->m_cell_size) { // time out
225 return -1;
226 }
227 }
228 } while (m_distance_to_next_pulse < m_vfo->m_cell_size);
229 // advance bit cell position
230 if (m_distance_to_next_pulse >= m_vfo->m_cell_size) {
231 m_distance_to_next_pulse -= m_vfo->m_cell_size;
232 }
233 return bit_reading;
234}

267bool mfm_codec::mfm_read_byte(uint8_t& data, bool& missing_clock, bool ignore_missing_clock, bool ignore_sync_field) {
268 size_t decode_count = 0;
269 missing_clock = false;
270
271 if (is_track_ready() == false) {
272 data = 0;
273 missing_clock = false;
274 return false;
275 }
276
277 do {
278 int bit_data = read_bit_ds();
279 if (bit_data == -1) {
280 data = 0;
281 missing_clock = false;
282 return false;
283 }
284 decode_count++;
285 if (m_track.is_wraparound()) {
286 m_track.clear_wraparound_flag();
287 m_wraparound = true;
288 }
289 m_bit_stream = (m_bit_stream << 1) | bit_data;
290
291 if (ignore_missing_clock == false && (decode_count & 0b01u)==0) { // Ignore MC on clock phase
292 if ((m_bit_stream & 0x0ffffu) == m_missing_clock_a1) { // Missing clock 0xA1 pattern
293 data = 0xa1;
294 missing_clock = true;
295 return true;
296 }
297 if ((m_bit_stream & 0x0ffffu) == m_missing_clock_c2) { // Missing clock 0xC2 pattern
298 data = 0xc2;
299 missing_clock = true;
300 return true;
301 }
302 }
303 if (ignore_sync_field == false) {
304 if (m_bit_stream == m_pattern_00) {
305 decode_count &= ~0b01u; // C/D synchronize
306 set_vfo_gain(mfm_codec::gain_state::high);
307 }
308 else {
309 set_vfo_gain(mfm_codec::gain_state::low);
310 }
311 }
312 } while (decode_count < 16);
313
314 // Extract only 'D' bits (exclude 'C' bits). MFM data is 'CDCDCDCD..CD'.
315 uint8_t read_data = 0;
316 for (uint16_t bit_pos = 0x4000; bit_pos > 0; bit_pos >>= 2) {
317 read_data = (read_data << 1) | ((m_bit_stream & bit_pos) ? 1 : 0);
318 }
319 data = read_data;
320 missing_clock = false;
321 return true;
322}

332uint16_t mfm_codec::mfm_encoder(uint8_t data, bool mode) {
333 // Data swap for special code
334 uint8_t write_data = data;
335 if (mode == true) { // special coding mode (F5 and F6 will be converted to A1* C2* with missing-clock pattern respectively)
336 switch (data) {
337 case 0xf5:
338 write_data = 0xa1;
339 break;
340 case 0xf6:
341 write_data = 0xc2;
342 break;
343 }
344 }
345 // MFM encoding
346 uint16_t mfm_bit_pattern = 0;
347 for (size_t bit_pos = 0x80; bit_pos != 0; bit_pos >>= 1) {
348 int current_bit = (write_data & bit_pos) ? 1 : 0;
349 int clock_bit = (m_prev_write_bit == 0 && current_bit == 0) ? 1 : 0;
350 mfm_bit_pattern = (mfm_bit_pattern << 1) | clock_bit; // clock
351 mfm_bit_pattern = (mfm_bit_pattern << 1) | current_bit; // data
352 m_prev_write_bit = current_bit;
353 }
354 // Missing clock operation
355 if (mode == true) {
356 switch (data) {
357 case 0xf5:
358 mfm_bit_pattern &= ~0b100000u; // remove a clock bit
359 break;
360 case 0xf6:
361 mfm_bit_pattern &= ~0b10000000u; // remove a clock bit
362 break;
363 }
364 }
365 return mfm_bit_pattern;
366}

376void mfm_codec::mfm_write_byte(uint8_t data, bool mode, bool write_gate, bool elastic) {
377 uint16_t bit_pattern = mfm_encoder(data, mode);
378 for (uint16_t bit_pos = 0x8000; bit_pos != 0; bit_pos >>= 1) {
379 int bit = (bit_pattern & bit_pos) ? 1 : 0;
380 if (write_gate == true) {
381 for (size_t i = 0; i < m_bit_width_w; i++) {
382 if (m_bit_width_w / 2 == i) m_track.write_stream(bit, elastic); // write data pulse at the center of the bit cell
383 else m_track.write_stream(0, elastic);
384 }
385 }
386 else {
387 for (size_t i = 0; i < m_bit_width_w; i++) {
388 m_track.advance_stream_pos(elastic); // advance stream pointer without actual data write (dummy write)
389 }
390 }
391 }
