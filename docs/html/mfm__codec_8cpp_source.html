67// ---------- Data separator
68
69// How data separator work:
70//
71// | WWWW | WWWW | WWWW | WWWW | WWWW | WWWW | WWWW | WWWW | <- Data window(W), Data cell boundary(|)
72// P P P P P <- Data pulses
73// 1 0 1 0 0 1 0 0 <- Data reading (only the data pulses within the data window will be considered)
74//
75// 01234567
76// | WWWW | <= In this case, bit cell size = 8, data window ofst = 2, data window size = 4
77//
78int mfm_codec::read_bit_ds(void) {
79 int bit_reading = 0;
80 double cell_center;
81 do {
82 // check if the next bit is within the next data window
83 if (m_distance_to_next_pulse < m_bit_cell_size) {
84 // check pulse position
85 if (m_distance_to_next_pulse >= m_data_window_ofst &&
86 // regular pulse (within the data window)
87 m_distance_to_next_pulse < m_data_window_ofst + m_data_window_size) {
88 bit_reading = 1;
89 }
90 else {
91 // irregular pulse
92#ifdef DEBUG
93 std::cout << '?';
94#endif
95 }
96 // adjust pulse phase (imitate PLL operation)
97 // limit the PLL operation frequency and introduce fluctuation with the random generator (certain fluctuation is required to reproduce some copy protection)
98 if (m_fluctuation == false || m_rnd() % m_fluctuator_denominator >= m_fluctuator_numerator) {
99 cell_center = m_data_window_ofst + m_data_window_size / 2.f;
100 double error = m_distance_to_next_pulse - cell_center;
101
102 // data pulse position adjustment == phase correction
103 m_distance_to_next_pulse -= error * 0.5;
104
105 // cell size adjustment == frequency correction
106 double new_cell_size = m_bit_cell_size + error * 0.1f;
107 // limit the range of cell size fluctuation
108 if (new_cell_size < m_bit_cell_size_ref * 0.8) new_cell_size = m_bit_cell_size_ref * 0.8;
109 if (new_cell_size > m_bit_cell_size_ref * 1.2) new_cell_size = m_bit_cell_size_ref * 1.2;
110 set_cell_size(new_cell_size);
111 }
112 size_t distance = m_track.distance_to_next_bit1();
113#if 0
114 // give timing fluctuation (intentionally - to immitate the spndle rotation fluctuation)
115 size_t rand_num = m_rnd() % 32;
116 if (rand_num == 0) {
117 distance++;
118 }
119 else if (rand_num == 1) {
120 distance--;
121 }
122#endif
123 m_distance_to_next_pulse += distance;
124 }
125 } while (m_distance_to_next_pulse < m_bit_cell_size);
126 // advance bit cell position
127 if (m_distance_to_next_pulse >= m_bit_cell_size) {
128 m_distance_to_next_pulse -= m_bit_cell_size;
129 }
130 return bit_reading;
131}

139void mfm_codec::mfm_read_byte(uint8_t& data, bool& missing_clock, bool ignore_missing_clock, bool ignore_sync_field) {
140 size_t decode_count = 0;
141 missing_clock = false;
142 do {
143 int bit_data = read_bit_ds();
144 if (m_track.is_wraparound()) {
145 m_track.clear_wraparound_flag();
146 m_wraparound = true;
147 }
148 m_bit_stream = (m_bit_stream << 1) | bit_data;
149
150 if (ignore_missing_clock == false) {
151 if ((m_bit_stream & 0x0ffffu) == m_missing_clock_a1) { // Missing clock 0xA1 pattern
152 data = 0xa1;
153 missing_clock = true;
154 return;
155 }
156 if ((m_bit_stream & 0x0ffffu) == m_missing_clock_c2) { // Missing clock 0xC2 pattern
157 data = 0xc2;
158 missing_clock = true;
159 return;
160 }
161 }
162 if (ignore_sync_field) {
163 if (m_bit_stream == m_pattern_00) {
164 m_sync_mode = true;
165 }
166 else {
167 m_sync_mode = false;
168 }
169 }
170 } while (++decode_count < 16);
171 uint8_t read_data = 0;
172 // Extract only 'D' bits (exclude 'C' bits). MFM data is 'CDCDCDCD..CD'.
173 for (uint16_t bit_pos = 0x4000; bit_pos > 0; bit_pos >>= 2) {
174 read_data = (read_data << 1) | (m_bit_stream & bit_pos ? 1 : 0);
175 }
176 data = read_data;
177 missing_clock = false;
178 return;
179}

182// mode: false=normal, true=special(write track, etc. cares 'F5' and 'F6')
183uint16_t mfm_codec::mfm_encoder(uint8_t data, bool mode) {
184 // Data swap for special code
185 uint8_t write_data = data;
186 if (mode == true) { // special coding mode (F5 and F6 will be converted to A1* C2* with missing-clock pattern respectively)
187 switch (data) {
188 case 0xf5:
189 write_data = 0xa1;
190 break;
191 case 0xf6:
192 write_data = 0xc2;
193 break;
194 }
195 }
196 // MFM encoding
197 uint16_t bit_pattern = 0;
198 uint16_t mfm_bit_pattern = 0;
199 int current_bit;
200 int clock_bit;
201 for (size_t bit_pos = 0x80; bit_pos != 0; bit_pos >>= 1) {
202 current_bit = (write_data & bit_pos) ? 1 : 0;
203 clock_bit = (m_prev_write_bit == 0 && current_bit == 0) ? 1 : 0;
204 mfm_bit_pattern = (mfm_bit_pattern << 1) | clock_bit; // clock
205 mfm_bit_pattern = (mfm_bit_pattern << 1) | current_bit; // data
206 m_prev_write_bit = current_bit;
207 }
208 // Missing clock operation
209 if (mode == true) {
210 switch (data) {
211 case 0xf5:
212 mfm_bit_pattern &= ~0b100000u; // remove a clock bit
213 break;
214 case 0xf6:
215 mfm_bit_pattern &= ~0b10000000u; // remove a clock bit
216 break;
217 }
218 }
219 return mfm_bit_pattern;
220}

222// mode: false=normal, true=special(write track, etc)
223// write_gate: true=perform actual write false=dummy write (no actual write will be perfored)
224void mfm_codec::mfm_write_byte(uint8_t data, bool mode, bool write_gate) {
225 uint16_t bit_pattern = mfm_encoder(data, mode);
226 for (uint16_t bit_pos = 0x8000; bit_pos != 0; bit_pos >>= 1) {
227 int bit = bit_pattern & bit_pos ? 1 : 0;
228 if (write_gate == true) {
229 for (size_t i = 0; i < m_bit_width; i++) {
230 if (m_bit_width / 2 == i) m_track.write_stream(bit); // write data pulse at the center of the bit cell
231 else m_track.write_stream(0);
232 }
233 }
234 else {
235 for (size_t i = 0; i < m_bit_width; i++) {
236 m_track.advance_stream_pos(); // advance stream pointer without actual data write (dummy write)
237 }
238 }
239 }
