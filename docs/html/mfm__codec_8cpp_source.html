137// How data separator work:
138//
139// | WWWW | WWWW | WWWW | WWWW | WWWW | WWWW | WWWW | WWWW | <- Data window(W), Data cell boundary(|)
140// P P P P P <- Data pulses
141// 1 0 1 0 0 1 0 0 <- Data reading (only the data pulses within the data window will be considered)
142//
143// 01234567
144// | WWWW | <= In this case, bit cell size = 8, data window ofst = 2, data window size = 4
145//

153int mfm_codec::read_bit_ds(void) {
154 int bit_reading = 0;
155 double cell_center;
156 size_t loop_count = 0; // for timeout check
157
158 if (is_track_ready() == false) {
159 return -1;
160 }
161
162 do {
163 // check if the next bit is within the next data window
164 if (m_distance_to_next_pulse < m_bit_cell_size) {
165 // check pulse position
166 if (m_distance_to_next_pulse >= m_data_window_ofst &&
167 // regular pulse (within the data window)
168 m_distance_to_next_pulse < m_data_window_ofst + m_data_window_size) {
169 bit_reading = 1;
170 }
171 else {
172 // irregular pulse
173#ifdef DEBUG
174 std::cout << '?';
175#endif
176 }
177 // Adjust pulse phase (imitate PLL operation)
178 // Limit the PLL operation frequency and introduce fluctuation with the random generator (certain fluctuation is required to reproduce some copy protection)
179 if (m_fluctuation == false || m_rnd() % m_fluctuator_denominator >= m_fluctuator_numerator) {
180 cell_center = m_data_window_ofst + m_data_window_size / 2.f;
181 double error = m_distance_to_next_pulse - cell_center;
182
183 // Data pulse position adjustment == phase correction
184 m_distance_to_next_pulse -= error * 0.05f * m_pll_gain;
185
186 // Cell size adjustment == frequency correction
187 double new_cell_size = m_bit_cell_size + error * 0.01f * m_pll_gain;
188 // Limit the range of cell size
189 if (new_cell_size < m_bit_cell_size_ref * 0.8) new_cell_size = m_bit_cell_size_ref * 0.8;
190 if (new_cell_size > m_bit_cell_size_ref * 1.2) new_cell_size = m_bit_cell_size_ref * 1.2;
191 set_cell_size(new_cell_size);
192 }
193 size_t distance = m_track.distance_to_next_bit1();
194#if 0
195 // give timing fluctuation (intentionally - to immitate the spndle rotation fluctuation)
196 size_t rand_num = m_rnd() % 32;
197 if (rand_num == 0) {
198 distance++;
199 }
200 else if (rand_num == 1) {
201 distance--;
202 }
203#endif
204 m_distance_to_next_pulse += distance;
205
206 if (loop_count++ > m_bit_cell_size) { // time out
207 return -1;
208 }
209 }
210 } while (m_distance_to_next_pulse < m_bit_cell_size);
211 // advance bit cell position
212 if (m_distance_to_next_pulse >= m_bit_cell_size) {
213 m_distance_to_next_pulse -= m_bit_cell_size;
214 }
215 return bit_reading;
216}

243bool mfm_codec::mfm_read_byte(uint8_t& data, bool& missing_clock, bool ignore_missing_clock, bool ignore_sync_field) {
244 size_t decode_count = 0;
245 missing_clock = false;
246
247 if (is_track_ready() == false) {
248 data = 0;
249 missing_clock = false;
250 return false;
251 }
252
253 do {
254 int bit_data = read_bit_ds();
255 if (bit_data == -1) {
256 data = 0;
257 missing_clock = false;
258 return false;
259 }
260 decode_count++;
261 if (m_track.is_wraparound()) {
262 m_track.clear_wraparound_flag();
263 m_wraparound = true;
264 }
265 m_bit_stream = (m_bit_stream << 1) | bit_data;
266
267 if (ignore_missing_clock == false) {
268 if ((m_bit_stream & 0x0ffffu) == m_missing_clock_a1) { // Missing clock 0xA1 pattern
269 data = 0xa1;
270 missing_clock = true;
271 return true;
272 }
273 if ((m_bit_stream & 0x0ffffu) == m_missing_clock_c2) { // Missing clock 0xC2 pattern
274 data = 0xc2;
275 missing_clock = true;
276 return true;
277 }
278 }
279 if (ignore_sync_field == false) {
280 if (m_bit_stream == m_pattern_00) {
281 decode_count &= ~0b01u; // C/D synchronize
282 set_gain(10.f);
283 }
284 else {
285 set_gain(1.f);
286 }
287 }
288 } while (decode_count < 16);
289
290 // Extract only 'D' bits (exclude 'C' bits). MFM data is 'CDCDCDCD..CD'.
291 uint8_t read_data = 0;
292 for (uint16_t bit_pos = 0x4000; bit_pos > 0; bit_pos >>= 2) {
293 read_data = (read_data << 1) | (m_bit_stream & bit_pos ? 1 : 0);
294 }
295 data = read_data;
296 missing_clock = false;
297 return true;
298}

308uint16_t mfm_codec::mfm_encoder(uint8_t data, bool mode) {
309 // Data swap for special code
310 uint8_t write_data = data;
311 if (mode == true) { // special coding mode (F5 and F6 will be converted to A1* C2* with missing-clock pattern respectively)
312 switch (data) {
313 case 0xf5:
314 write_data = 0xa1;
315 break;
316 case 0xf6:
317 write_data = 0xc2;
318 break;
319 }
320 }
321 // MFM encoding
322 uint16_t bit_pattern = 0;
323 uint16_t mfm_bit_pattern = 0;
324 int current_bit;
325 int clock_bit;
326 for (size_t bit_pos = 0x80; bit_pos != 0; bit_pos >>= 1) {
327 current_bit = (write_data & bit_pos) ? 1 : 0;
328 clock_bit = (m_prev_write_bit == 0 && current_bit == 0) ? 1 : 0;
329 mfm_bit_pattern = (mfm_bit_pattern << 1) | clock_bit; // clock
330 mfm_bit_pattern = (mfm_bit_pattern << 1) | current_bit; // data
331 m_prev_write_bit = current_bit;
332 }
333 // Missing clock operation
334 if (mode == true) {
335 switch (data) {
336 case 0xf5:
337 mfm_bit_pattern &= ~0b100000u; // remove a clock bit
338 break;
339 case 0xf6:
340 mfm_bit_pattern &= ~0b10000000u; // remove a clock bit
341 break;
342 }
343 }
344 return mfm_bit_pattern;
345}

355void mfm_codec::mfm_write_byte(uint8_t data, bool mode, bool write_gate) {
356 if (is_track_ready() == false) {
357 return;
358 }
359 uint16_t bit_pattern = mfm_encoder(data, mode);
360 for (uint16_t bit_pos = 0x8000; bit_pos != 0; bit_pos >>= 1) {
361 int bit = bit_pattern & bit_pos ? 1 : 0;
362 if (write_gate == true) {
363 for (size_t i = 0; i < m_bit_width; i++) {
364 if (m_bit_width / 2 == i) m_track.write_stream(bit); // write data pulse at the center of the bit cell
365 else m_track.write_stream(0);
366 }
367 }
368 else {
369 for (size_t i = 0; i < m_bit_width; i++) {
370 m_track.advance_stream_pos(); // advance stream pointer without actual data write (dummy write)
371 }
372 }
373 }
374}
