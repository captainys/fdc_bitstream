88std::vector<size_t> find_peaks(std::vector<size_t> vec) {
89 // calc moving average
90 vec.resize(vec.size()+2); // increase top boundary for moving average calculation
91 std::vector<size_t> avg(vec.size());
92 for(size_t i = 2; i<vec.size()-2-1; i++) {
93 size_t avg_tmp = 0;
94 for(int j=-2; j<=2; j++) {
95 avg_tmp += vec[i+j];
96 }
97 avg[i] = avg_tmp / 5;
98 }
99 //display_histogram(avg); // for debug purpose
100
101 // detect peaks
102 std::vector<std::pair<size_t, size_t>> peaks;
103 for(size_t i=1; i<avg.size()-1-1; i++) {
104 if(avg[i-1]<avg[i] && avg[i+1]<avg[i]) { // peak
105 peaks.push_back(std::pair<size_t, size_t>(i, avg[i]));
106 }
107 }
108
109 // sort peaks
110 std::sort(peaks.begin(), peaks.end(), [](const std::pair<size_t, size_t> &left, const std::pair<size_t, size_t> &right)
111 { return left.second > right.second; });
112
113 // Display sort result
114 //for(auto it=peaks.begin(); it != peaks.end(); ++it) {
115 // std::cout << (*it).first << "," << (*it).second << std::endl;
116 //}
117
118 if (peaks.size()<3) {
119 peaks.resize(3);
120 }
121 std::vector<size_t> result;
122 for(auto it=peaks.begin(); it != peaks.begin()+3; ++it) {
123 result.push_back((*it).first);
124 }
125
126 std::sort(result.begin(), result.end(), [](const size_t &left, const size_t &right)
127 { return left < right; });
128
129 return result;

143std::vector<bit_array> normalize_track(std::vector<bit_array> tracks, size_t sampling_rate, size_t bit_rate) {
144 std::cout << "=== Pulse pitch normalize ===" << std::endl;

149 std::vector<bit_array> res;
150 bit_array trk;
151 for(size_t i=0; i<tracks.size(); i++) {
152 trk = tracks[i];
153 bit_array normalized;

158 //display_histogram(freq_dist);

160 double cell_size;

164 cell_size = cell_size_std; // to avoid 0div
