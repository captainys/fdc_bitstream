62std::vector<bit_array> normalize_track(std::vector<bit_array> tracks, size_t sampling_rate, size_t bit_rate, bool verbose=false) {
63 if(verbose) {
64 std::cout << "=== Pulse pitch normalize ===" << std::endl;
65 std::cout << "Sampling rate : " << sampling_rate / 1e6 << " MHz" << std::endl;
66 std::cout << "Data bit rate : " << bit_rate / 1e3 << " Kbits/sec" << std::endl;
67 }
68 double cell_size_std = static_cast<double>(sampling_rate) / static_cast<double>(bit_rate);
69
70 std::vector<bit_array> res;
71 bit_array trk;
72 for(size_t i=0; i<tracks.size(); i++) {
73 trk = tracks[i];
74 bit_array normalized;
75 normalized.clear_array();
76 if(trk.get_length() > 0) {
77 if (verbose) {
78 std::cout << "Track " << std::setw(3) << i << " - ";
79 }
80 std::vector<size_t> freq_dist = fdc_misc::get_frequent_distribution(trk); // convert track data into frequency-distribution data
81 //display_histogram(freq_dist);
82 std::vector<size_t> peaks = fdc_misc::find_peaks(freq_dist); // find peaks from the frequency-distribution data
83 double cell_size;
84 if(peaks[0] != 0) {
85 cell_size = static_cast<double>(peaks[0]) / 2.f;
86 } else {
87 cell_size = cell_size_std; // to avoid 0div
88 }
89 double scale = cell_size_std / cell_size;
90 if(verbose) {
91 std::cout << "Scale : " << std::fixed << std::setprecision(5) << scale;
92 std::cout << ", Peaks : (" << peaks[0] << ", " << peaks[1] << ", " << peaks[2] << ")" << std::endl;
93 }
94 std::vector<size_t> dist_array = fdc_misc::convert_to_dist_array(trk);
95 double curr_pos = 0.f;
96 for(auto it = dist_array.begin(); it != dist_array.end(); ++it) {
97 curr_pos += (*it) * scale;
98 normalized.set(curr_pos, 1);
99 }
100 }
101 res.push_back(normalized);
102 }
103 return res;
104}

117void inspect_track(std::vector<bit_array> &track_data, std::vector<std::unordered_map<std::string, std::size_t>> &inspect_result,
118 const size_t vfo_type, const double gain_l, const double gain_h,
119 const size_t sampling_rate, const size_t data_bit_rate, bool verbose) {
120 fdc_bitstream fdc;
121
122 size_t cell_size_ref = sampling_rate / data_bit_rate;
123 size_t sect_pos_ofst = (cell_size_ref * 16) * 16; // sector read start position needs to be ahead a bit from the sector ID start position.
124
125 inspect_result.resize(track_data.size());
126 for(size_t track_n = 0; track_n < track_data.size(); track_n++) {
127 fdc.set_track_data(track_data[track_n]);
128 fdc.set_vfo_type(vfo_type);
129 fdc.set_vfo_gain_val(gain_l, gain_h);
130 fdc.set_vfo_gain_mode(fdc_bitstream::gain_state::low);
131 std::vector<fdc_bitstream::id_field> id_list = fdc.read_all_idam(); // Read all IDAMs
132 inspect_result[track_n]["num_ids"] = id_list.size();
133 size_t sector_good = 0;
134 size_t sector_bad = 0;
135 for(auto id = id_list.begin(); id != id_list.end(); id++) { // Read all sectors
136 // read and inspect a sector
137 size_t sect_pos = (*id).pos;
138 sect_pos = (sect_pos < sect_pos_ofst) ? 0 : sect_pos - sect_pos_ofst;
139 fdc.set_pos(sect_pos);
140 fdc_bitstream::sector_data read_sect = fdc.read_sector((*id).C, (*id).R);
141 if (!read_sect.record_not_found && !read_sect.crc_sts && !(*id).crc_sts) {
142 sector_good++;
143 } else {
144 sector_bad++;
145 }
146 }
147 inspect_result[track_n]["num_good"] = sector_good;
148 inspect_result[track_n]["num_bad"] = sector_bad;
149 if(verbose) std::cout << "." << std::flush;
150 }
