18void bit_array::set_array(std::vector<uint8_t>& buf) {
19 m_array_data.resize(buf.size()*8);
20 size_t pos = 0;
21 for (auto it = buf.begin(); it != buf.end(); ++it) {
22 uint8_t dt = *it;
23 for (uint8_t bit_mask = 0x80; bit_mask != 0; bit_mask >>= 1) {
24 uint8_t bit = (dt & bit_mask) ? 1 : 0;
25 m_array_data[pos++] = bit;
26 }
27 }
28 m_stream_pos = 0;
29}

47std::vector<uint8_t> bit_array::get_array(void) {
48 std::vector<uint8_t> result;
49 result.resize(m_array_data.size()/8 + ((m_array_data.size()%8)?1:0));
50 uint8_t bit_ptn = 0x80;
51 uint8_t dt = 0;
52 size_t pos = 0;
53 for (auto it = m_array_data.begin(); it != m_array_data.end(); ++it) {
54 if (bit_ptn == 0) {
55 bit_ptn = 0x80;
56 }
57 dt |= *it ? bit_ptn : 0;
58 bit_ptn >>= 1;
59 result[pos++] = dt;
60 }
61 if (bit_ptn) {
62 result[pos++] = dt;
63 }
64 return result;
65}

278void bit_array::load(std::string file_name) {
279 std::ifstream ifs;
280 ifs.open(file_name, std::ios::in | std::ios::binary);
281 if (ifs.is_open() == false) {
282#ifdef DEBUG
283 std::cerr << "Failed to open a file'" << file_name << "'." << std::endl;
284#endif
285 clear_array();
286 return;
287 }
288 ifs.seekg(0, std::ios_base::end);
289 size_t size = ifs.tellg();
290 ifs.seekg(0, std::ios_base::beg);
291 size -= sizeof(size_t);
292 set_array(size);
293 ifs.read(reinterpret_cast<char*>(m_array_data.data()), size);
294 m_stream_pos = 0;
295}
