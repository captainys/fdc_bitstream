21void bit_array::set_array(std::vector<uint8_t>& buf) {
22 m_array_data.resize(buf.size()*8);
23 size_t pos = 0;
24 for (auto it = buf.begin(); it != buf.end(); ++it) {
25 uint8_t dt = *it;
26 for (uint8_t bit_mask = 0x80; bit_mask != 0; bit_mask >>= 1) {
27 uint8_t bit = (dt & bit_mask) ? 1 : 0;
28 m_array_data[pos++] = bit;
29 }
30 }
31 m_stream_pos = 0;
32}

50std::vector<uint8_t> bit_array::get_array(void) {
51 std::vector<uint8_t> result;
52 result.resize(m_array_data.size()/8 + ((m_array_data.size()%8)?1:0));
53 uint8_t bit_ptn = 0x80;
54 uint8_t dt = 0;
55 size_t pos = 0;
56 for (auto it = m_array_data.begin(); it != m_array_data.end(); ++it) {
57 dt |= *it ? bit_ptn : 0;
58 bit_ptn >>= 1;
59 if (bit_ptn == 0) {
60 bit_ptn = 0x80;
61 result[pos++] = dt;
62 dt = 0;
63 }
64 }
65 if (bit_ptn != 0x80) {
66 result[pos++] = dt;
67 }
68 return result;
69}

291void bit_array::load(std::string file_name) {
292 std::ifstream ifs;
293 ifs.open(file_name, std::ios::in | std::ios::binary);
294 if (ifs.is_open() == false) {
295#ifdef DEBUG
296 std::cerr << "Failed to open a file'" << file_name << "'." << std::endl;
297#endif
298 clear_array();
299 return;
300 }
301 ifs.seekg(0, std::ios_base::end);
302 size_t size = ifs.tellg();
303 ifs.seekg(0, std::ios_base::beg);
304 size -= sizeof(size_t);
305 set_array(size);
306 ifs.read(reinterpret_cast<char*>(m_array_data.data()), size);
307 m_stream_pos = 0;
308}
