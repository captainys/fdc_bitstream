20void bit_array::set_array(std::vector<uint8_t>& buf) {
21 m_array_data.resize(buf.size()*8);
22 size_t pos = 0;
23 for (auto it = buf.begin(); it != buf.end(); ++it) {
24 uint8_t dt = *it;
25 for (uint8_t bit_mask = 0x80; bit_mask != 0; bit_mask >>= 1) {
26 uint8_t bit = (dt & bit_mask) ? 1 : 0;
27 m_array_data[pos++] = bit;
28 }
29 }
30 m_stream_pos = 0;
31}

49std::vector<uint8_t> bit_array::get_array(void) {
50 std::vector<uint8_t> result;
51 result.resize(m_array_data.size()/8 + ((m_array_data.size()%8)?1:0));
52 uint8_t bit_ptn = 0x80;
53 uint8_t dt = 0;
54 size_t pos = 0;
55 for (auto it = m_array_data.begin(); it != m_array_data.end(); ++it) {
56 dt |= *it ? bit_ptn : 0;
57 bit_ptn >>= 1;
58 if (bit_ptn == 0) {
59 bit_ptn = 0x80;
60 result[pos++] = dt;
61 dt = 0;
62 }
63 }
64 if (bit_ptn != 0x80) {
65 result[pos++] = dt;
66 }
67 return result;
68}

290void bit_array::load(std::string file_name) {
291 std::ifstream ifs;
292 ifs.open(file_name, std::ios::in | std::ios::binary);
293 if (ifs.is_open() == false) {
294#ifdef DEBUG
295 std::cerr << "Failed to open a file'" << file_name << "'." << std::endl;
296#endif
297 clear_array();
298 return;
299 }
300 ifs.seekg(0, std::ios_base::end);
301 size_t size = ifs.tellg();
302 ifs.seekg(0, std::ios_base::beg);
303 size -= sizeof(size_t);
304 set_array(size);
305 ifs.read(reinterpret_cast<char*>(m_array_data.data()), size);
306 m_stream_pos = 0;
307}
