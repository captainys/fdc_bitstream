19void bit_array::set_array(std::vector<uint8_t>& buf) {
20 m_array_data.resize(buf.size()*8);
21 size_t pos = 0;
22 for (auto it = buf.begin(); it != buf.end(); ++it) {
23 uint8_t dt = *it;
24 for (uint8_t bit_mask = 0x80; bit_mask != 0; bit_mask >>= 1) {
25 uint8_t bit = (dt & bit_mask) ? 1 : 0;
26 m_array_data[pos++] = bit;
27 }
28 }
29 m_stream_pos = 0;

48std::vector<uint8_t> bit_array::get_array(void) {
49 std::vector<uint8_t> result;
50 result.resize(m_array_data.size()/8 + ((m_array_data.size()%8)?1:0));
51 uint8_t bit_ptn = 0x80;
52 uint8_t dt = 0;
53 size_t pos = 0;
54 for (auto it = m_array_data.begin(); it != m_array_data.end(); ++it) {
55 dt |= *it ? bit_ptn : 0;
56 bit_ptn >>= 1;
57 if (bit_ptn == 0) {
58 bit_ptn = 0x80;
59 result[pos++] = dt;
60 }
61 }
62 if (bit_ptn != 0x80) {
63 result[pos++] = dt;
64 }
65 return result;
66}

288void bit_array::load(std::string file_name) {
289 std::ifstream ifs;
290 ifs.open(file_name, std::ios::in | std::ios::binary);
291 if (ifs.is_open() == false) {
292#ifdef DEBUG
293 std::cerr << "Failed to open a file'" << file_name << "'." << std::endl;
294#endif
295 clear_array();
296 return;
297 }
298 ifs.seekg(0, std::ios_base::end);
299 size_t size = ifs.tellg();
300 ifs.seekg(0, std::ios_base::beg);
301 size -= sizeof(size_t);
302 set_array(size);
303 ifs.read(reinterpret_cast<char*>(m_array_data.data()), size);
304 m_stream_pos = 0;
305}
