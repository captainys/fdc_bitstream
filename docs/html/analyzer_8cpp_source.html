58 std::vector<std::string> allowed = { "hfe", "mfm", "raw", "d77" };

162void trim_track(bit_array &in_array, bit_array &out_array, size_t start_byte, size_t end_byte) {
163 constexpr size_t err = 99999999;
164 size_t start_pos=err, end_pos=err, prev_pos;
165 size_t read_count = 0;
166 fdc->set_track_data(in_array);
167 fdc->set_pos(0);
168 uint8_t dt;
169 bool mc;
170 std::cout << start_byte << " " << end_byte << std::endl;
171 do {
172 prev_pos = fdc->get_pos();
173 fdc->read_data(dt, mc, false, false);
174 if(read_count == start_byte) start_pos = prev_pos;
175 if(read_count == end_byte) end_pos = fdc->get_pos();
176 read_count++;
177 } while(!fdc->is_wraparound());
178 if(start_pos == err || end_pos == err) {
179 std::cout << "Could not find start or end position." << std::endl;
180 out_array = in_array;
181 return;
182 }
183 out_array.clear_array();
184 in_array.set_stream_pos(start_pos);
185 for(size_t pos = start_pos; pos<end_pos; pos++) {
186 out_array.write_stream(in_array.read_stream(), true);
187 }

256void cmd_read_sector(size_t cyl, size_t rcd, bool pulse_vis) {
257 if(is_image_ready()==false) {
258 std::cout << "Disk image is not ready." << std::endl;
259 return;
260 }
261 bit_array track_stream;
262 fdc_bitstream::sector_data read_data;
263 track_stream = disk_img->get_track_data(cyl);
264 fdc->set_track_data(track_stream);
265 if(rcd >= 1000) { // read by sector index #
266 std::vector<fdc_bitstream::id_field> id_list = fdc->read_all_idam(); // get the list of all IDs
267 size_t sct_idx = rcd - 1000;
268 if (sct_idx >= id_list.size()) {
269 std::cout << "Index exceeded (the number of sectors)-1 in the track. (" << sct_idx << ">" << id_list.size()-1 << ")." << std::endl;
270 return;
271 }
272 fdc_bitstream::id_field sct_id = id_list[sct_idx];
273 std::cout << "Sector read by index : (" << std::dec << sct_idx + 1 << ") " << std::hex
274 << std::setfill('0')
275 << std::setw(2) << static_cast<int>(sct_id.C) << " " << std::setw(2) << static_cast<int>(sct_id.H) << " "
276 << std::setw(2) << static_cast<int>(sct_id.R) << " " << std::setw(2) << static_cast<int>(sct_id.N) << std::dec << std::endl;
277 size_t cell_size = g_sampling_rate / g_data_bit_rate;
278 size_t rewind = (cell_size * 16 /*1byte*/) * 16;
279 size_t pos = sct_id.pos;
280 pos = (pos >= rewind) ? pos-rewind : 0;
281 cyl = sct_id.C;
282 rcd = sct_id.R;
283 fdc->set_pos(pos); // rewind the start position a bit.
284 } else {
285 std::cout << "Sector read : " << cyl << ", " << rcd << std::endl;
286 cyl /= 2;
287 }
288 read_data = fdc->read_sector(cyl, rcd);
289 if(true!=pulse_vis)
290 {
291 // hex dump
292 fdc_misc::dump_buf(read_data.data.data(), read_data.data.size(), true, 16, 16, true);
293 }
294 else
295 {
296 // bit dump
297 for(size_t i=0; i<read_data.data.size() && i<read_data.pos.size(); ++i)
298 {
299 if(i % 8==0) { // display standard bit cell guide (ruler)
300 size_t standard_bit_cell = g_sampling_rate / g_data_bit_rate;
301 std::cout << std::string(19, ' ');
302 for(size_t ii=0; ii<8; ii++) {
303 std::cout << "|" << std::string(standard_bit_cell-1, 'c');

305 }
306 std::cout << std::endl;
307 }
308 size_t pulse_end=(i+1<read_data.pos.size() ? read_data.pos[i+1] : read_data.data_end_pos);
309 std::cout << "+" << std::hex << std::setw(2) << i << ":";
310 std::cout << std::hex << std::setw(2) << std::setfill('0') << int(read_data.data[i]);
311 std::cout << std::dec << std::setw(10) << std::setfill(' ') << read_data.pos[i];
312 std::cout << " ";

315 std::cout << (int(track_stream.get(j)) ? "#":".");
316 }
317 std::cout << std::endl;
318 }
319 }
320 size_t end_pos = fdc->get_pos();
321 if (end_pos < read_data.id_pos) {
322 end_pos += track_stream.get_length(); // wrap around correction
323 }
324 double read_time = (static_cast<double>(end_pos - read_data.id_pos)*1000.f) / g_sampling_rate;
325 std::cout << "CRC DAM RNF ----ID_POS ---DAM_POS ---END_POS ---TIME(ms) SIZE" << std::endl;
326 std::cout << (read_data.crc_sts ? "ERR " : "OK ") << (read_data.dam_type ? "DDAM " : "DAM ") << (read_data.record_not_found ? "ERR " : "OK ");
327 std::cout << std::setfill(' ');
328 std::cout << std::setw(10) << read_data.id_pos << " " << std::setw(10) << read_data.data_pos << " " <<
329 std::setw(10) << end_pos << " " << std::setw(10) << std::setprecision(10) << read_time << " " << read_data.data.size() << std::endl;
330}

346void cmd_visualize_vfo(size_t track_n, size_t vfo_sel=99) {
347 if(is_image_ready()==false) {
348 std::cout << "Disk image is not ready." << std::endl;
349 return;
350 }
351 bit_array track_stream;
352 track_stream = disk_img->get_track_data(track_n);
353 vfo_base *vfo;
354 if(vfo_sel == 99) {
355 vfo_sel = g_vfo_type;
356 }
357 switch(vfo_sel) {
358 default:
359 case VFO_TYPE_SIMPLE: vfo = new vfo_simple(); break;
360 case VFO_TYPE_FIXED: vfo = new vfo_fixed(); break;
361 case VFO_TYPE_PID: vfo = new vfo_pid(); break;
362 case VFO_TYPE_PID2: vfo = new vfo_pid2(); break;
363 case VFO_TYPE_SIMPLE2: vfo = new vfo_simple2(); break;
364 case VFO_TYPE_PID3: vfo = new vfo_pid3(); break;
365 case VFO_TYPE_EXPERIMANTAL: vfo = new vfo_experimental(); break;
366 }
367
368 disk_image_base_properties props = disk_img->get_property();
369 vfo->set_params(props.m_sampling_rate, props.m_data_bit_rate);
370 vfo->set_gain_val(g_gain_l, g_gain_h);
371 vfo->set_gain_mode(vfo_base::gain_state::low);
372 track_stream.set_stream_pos(0);
373
374 double scale = (150 * 0.5f) / (vfo->m_cell_size_ref);
375
376 double dist = -1.f;
377 size_t irregular_pulse_count = 0;
378 size_t count = 0;
379 bool irregular = false;
380
381#ifdef _WIN32
382 while(count<500000 && !track_stream.is_wraparound()) {
383#else
384 while(count<5000 && !track_stream.is_wraparound()) {
385#endif
386 irregular = false;
387
388 std::string line = std::string(150, ' ');
389 size_t win_st = (vfo->m_window_ofst) * scale;
390 size_t win_en = (vfo->m_window_ofst+vfo->m_window_size) * scale;
391 for(size_t x = win_st; x < win_en; x++) line[x] = '-';
392 line[vfo->m_cell_size * scale] = '<';
393
394 do {
395 if (dist < vfo->m_cell_size) {
396 // visualize
397 if(dist<0.f) line[0] = '*';
398 else if(dist * scale >= line.size()) line[vfo->m_cell_size*scale] = '*';
399 else line[dist * scale] = 'P';
400 if(dist < vfo->m_window_ofst || dist > vfo->m_window_ofst + vfo->m_window_size) {
401 irregular = true;
402 irregular_pulse_count++;
403 }
404 // run VFO
405 dist = vfo->calc(dist);
406 dist += static_cast<double>(track_stream.distance_to_next_pulse());
407 }
408 } while (dist < vfo->m_cell_size);
409 dist -= vfo->m_cell_size;
410
411 std::cout << std::setw(6) << count << " " << std::setw(6) << std::setprecision(4) << std::setfill(' ') << vfo->m_cell_size << " >" << line;
412 if(irregular) {
413 fdc_misc::color(6);
414 std::cout << "\a*** IRREGULAR PULSE DETECTED ***";
415 irregular_pulse_count++;
416 fdc_misc::color(7);
417 }
418 std::cout << std::endl;
419 count++;
420#if _WIN32
421 if(kbhit()) {
422 getch();
423 break;
424 }
425#endif
426 }
427 vfo->disp_vfo_status();
428 fdc_misc::color(4);
429 std::cout << irregular_pulse_count << " irregular pulse(s) detected." << std::endl;
430 fdc_misc::color(7);
431
432 delete vfo;
433}

481void cmd_normalize_track(size_t track_n) {
482 if(is_image_ready()==false) {
483 std::cout << "Disk image is not ready." << std::endl;
484 return;
485 }
486
487 size_t track_n_s, track_n_e;
488 if (track_n == 65535) { // target = all track
489 track_n_s = 0;
490 track_n_e = disk_img->get_number_of_tracks()-1;
491 } else {
492 track_n_s = track_n_e = track_n;
493 }
494
495 bit_array track_stream;
496 bit_array normalized_track;
497 disk_image_base_properties props = disk_img->get_property();
498
499 double cell_size_ref = static_cast<double>(props.m_sampling_rate) / static_cast<double>(props.m_data_bit_rate);
500 double cell_size = cell_size_ref;
501 double new_cell_size;
502 std::cout << "Standard cell size : " << cell_size_ref << std::endl;
503
504 constexpr int window_size = 7;
505 for(size_t trk_n = track_n_s; trk_n <= track_n_e; trk_n++) {
506 track_stream = disk_img->get_track_data(trk_n);
507 track_stream.set_stream_pos(0);
508 normalized_track.clear_array();
509 size_t current_pos = 0;
510 std::cout << trk_n << ", " << std::flush;
511 std::vector<double> dist_array;
512 while(!track_stream.is_wraparound()) {
513 dist_array.push_back(static_cast<double>(track_stream.distance_to_next_pulse()));
514 }
515 // calculate moving average of cell_size (window size = window_size)
516 for(int i = 0; i < dist_array.size(); i++) {
517 double mov_avg = 0.f;
518 for(int j = -window_size/2; j <= window_size/2; j++) {
519 double tmp_cell_size, tmp_dist;
520 if(i+j < 0) {
521 tmp_dist = dist_array[-i-j]; // == [-(i+j)]; // mirroring on lower boundary
522 } else if(i+j >= dist_array.size()) {
523 tmp_dist = dist_array[dist_array.size()*2 - i - j - 1]; // == [dist_array.size()-((i+j)-dist_array.size()+1)] // mirroring on upper boudary
524 } else {
525 tmp_dist = dist_array[i+j];
526 }
527 tmp_cell_size = round(tmp_dist/cell_size); // quantize (by current cell size)
528 if(tmp_cell_size>=2 && tmp_cell_size <=4) {
529 mov_avg += tmp_dist / tmp_cell_size; // add quantized cell size if it's normal size
530 } else {
531 mov_avg += cell_size; // ignore outlier data (use current cell size instead)
532 }
533 }
534 mov_avg /= static_cast<double>(window_size); // moving average of cell_size
535
536 double dist = dist_array[i];
537 size_t quantized = round(dist/mov_avg); // quantize
538
539 //if(quantized==1) quantized = 2;
540 //if(quantized>=5) quantized = 4;
541 current_pos += quantized * cell_size_ref;
542 normalized_track.set(current_pos, 1);
543
544 cell_size = mov_avg;
545 }
