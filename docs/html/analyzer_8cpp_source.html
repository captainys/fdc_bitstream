32void bit_dump(const uint64_t data, size_t bit_width, size_t spacing = 0, bool line_feed = true) {
33 size_t count = 0;
34 for (uint64_t bit_pos = 1 << (bit_width - 1); bit_pos != 0; bit_pos >>= 1) {
35 std::cout << ((data & bit_pos) ? 1 : 0);
36 count++;
37 if (spacing > 0) {
38 if (count % spacing == 0) {
39 std::cout << " ";
40 }
41 }
42 }
43 if (line_feed == true) {
44 std::cout << std::endl;

199void cmd_visualize_vfo(size_t track_n, size_t vfo_sel=0) {
200 if(is_image_ready()==false) {
201 std::cout << "Disk image is not ready." << std::endl;
202 return;
203 }
204 bit_array track_stream;
205 track_stream = disk_img->get_track_data(track_n);
206 vfo_base *vfo;
207 switch(vfo_sel) {
208 default:
209 case 0:
210 vfo = new vfo_simple();
211 break;
212 case 1:
213 vfo = new vfo_fixed();
214 break;
215 case 2:
216 vfo = new vfo_pid();
217 break;
218 case 3:
219 vfo = new vfo_pid2();
220 break;
221 }
222
223 disk_image_base_properties props = disk_img->get_property();
224 vfo->set_params(props.m_sampling_rate, props.m_data_bit_rate);
225 vfo->set_gain_val(g_gain_l, g_gain_h);
226 vfo->set_gain_mode(vfo_base::gain_state::low);
227 track_stream.set_stream_pos(0);
228
229 double dist = 0.f;
230 for(size_t i=0; i<5000; i++) {
231 dist += static_cast<double>(track_stream.distance_to_next_bit1());
232 //dist -= std::floor(dist / vfo->m_cell_size) * vfo->m_cell_size;
233 while(dist > vfo->m_cell_size) {
234 dist -= vfo->m_cell_size;
235 }
236
237 // visualize
238 std::string line = std::string(80, ' ');
239 for(size_t x = vfo->m_window_ofst; x<vfo->m_window_ofst+vfo->m_window_size; x++) {
240 line[x] = '_';
241 }
242 line[vfo->m_cell_size+1] = '<';
243 if(dist>=0.f) {
244 line[dist] = 'P';
245 } else {
246 line[0] = '*';
247 }
248 std::cout << line << std::endl;
249
250 // run VFO
251 dist = vfo->calc(dist);
252 }
